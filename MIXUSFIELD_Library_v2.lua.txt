-- Active theme data (starts as Crimson; changed via SetTheme)
local Theme = {}
for k, v in pairs(THEMES.Crimson) do Theme[k] = v end

-- Every object that should react to theme changes registers here.
-- Format: { Instance = guiObject, Property = "BackgroundColor3", Key = "Accent" }
local ThemeListeners = {}

-- Registers a GUI property to update whenever the theme changes.
-- @param inst   GuiObject to update
-- @param prop   Property name string (e.g. "BackgroundColor3")
-- @param key    Theme key string (e.g. "Accent")
local function ThemeBind(inst, prop, key)
  table.insert(ThemeListeners, { Instance = inst, Property = prop, Key = key })
  -- Apply immediately so the element is correctly colored at creation.
  if inst and inst.Parent then
    inst[prop] = Theme[key]
  end
end

local Custom = {}

-- Shorthand: create any Instance, set properties, optionally parent it.
-- @param Name       ClassName string
-- @param Properties key-value table of properties
-- @param Parent     optional parent Instance
function Custom:Create(Name, Properties, Parent)
  local inst = Instance.new(Name)
  for i, v in pairs(Properties) do
    inst[i] = v
  end
  if Parent then inst.Parent = Parent end
  return inst
end

-- Helper: resolve the safest parent for any ScreenGui (exploit or Studio context).
local function GetGuiParent()
  if RunService:IsStudio() then
    return Player.PlayerGui
  end
  return (pcall(function() return gethui() end) and gethui())
    or (pcall(function() return cloneref(game:GetService("CoreGui")) end) and cloneref(game:GetService("CoreGui")))
    or game:GetService("CoreGui")
end

-- ============================================================
-- /// RIPPLE EFFECT
-- Plays a material-design circle ripple on any button/frame.
-- @param Button  The GuiObject to ripple inside
-- @param X, Y   Screen coordinates of the click/touch
-- ============================================================
local function CircleClick(Button, X, Y)
  task.spawn(function()
    Button.ClipsDescendants = true

    local Circle = Instance.new("ImageLabel")
    Circle.Image             = "rbxassetid://106471194043211"
    Circle.ImageColor3       = Color3.fromRGB(80, 80, 80)
    Circle.ImageTransparency = 0.9
    Circle.BackgroundTransparency = 1
    Circle.ZIndex            = 10
    Circle.Name              = "Ripple"
    Circle.Parent            = Button

    -- Position the circle at the click point relative to the button.
    local nx = X - Button.AbsolutePosition.X
    local ny = Y - Button.AbsolutePosition.Y
    Circle.Position = UDim2.new(0, nx, 0, ny)

    local Size = math.max(Button.AbsoluteSize.X, Button.AbsoluteSize.Y) * 1.5
    local info = TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)

    TweenService:Create(Circle, info, {
      Size     = UDim2.new(0, Size, 0, Size),
      Position = UDim2.new(0.5, -Size / 2, 0.5, -Size / 2),
    }):Play()

    task.wait(0.45)

    -- Fade out the circle after the expand finishes.
    for i = 1, 10 do
      Circle.ImageTransparency = Circle.ImageTransparency + 0.01
      task.wait(0.05)
    end
    Circle:Destroy()
  end)
end

-- ============================================================
-- /// DRAGGABLE HELPER
-- Makes `object` draggable by clicking and dragging `topbar`.
-- ============================================================
local function MakeDraggable(topbar, object)
  local dragging, dragStart, startPos = false, nil, nil

  topbar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
      dragging  = true
      dragStart = input.Position
      startPos  = object.Position

      input.Changed:Connect(function()
        if input.UserInputState == Enum.UserInputState.End then
          dragging = false
        end
      end)
    end
  end)

  topbar.InputChanged:Connect(function(input)
    if dragging and (
      input.UserInputType == Enum.UserInputType.MouseMovement
      or input.UserInputType == Enum.UserInputType.Touch
    ) then
      local delta = input.Position - dragStart
      object.Position = UDim2.new(
        startPos.X.Scale, startPos.X.Offset + delta.X,
        startPos.Y.Scale, startPos.Y.Offset + delta.Y
      )
    end
  end)
end

-- ============================================================
-- /// THEME SYSTEM – PUBLIC API
-- Call MIXUSFIELD_Library:SetTheme("Ocean") from any script.
-- You can also pass a custom table of colors.
-- ============================================================

-- Sets the active theme and re-colors every registered element.
-- @param nameOrTable  string preset name  OR  custom { Accent=…, … } table
function MIXUSFIELD_Library:SetTheme(nameOrTable)
  local newTheme

  if type(nameOrTable) == "string" then
    -- Use a built-in preset by name (case-insensitive lookup).
    for k, v in pairs(THEMES) do
      if k:lower() == nameOrTable:lower() then
        newTheme = v
        break
      end
    end
    if not newTheme then
      warn("[MIXUSFIELD] Unknown theme: " .. tostring(nameOrTable))
      return
    end
  elseif type(nameOrTable) == "table" then
    -- Accept a fully or partially custom color table.
    newTheme = nameOrTable
  else
    warn("[MIXUSFIELD] SetTheme expects a string or color table.")
    return
  end

  -- Copy new values into the shared Theme table.
  for k, v in pairs(newTheme) do
    Theme[k] = v
  end

  -- Propagate to every registered element.
  local tweenInfo = TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
  for _, entry in ipairs(ThemeListeners) do
    local ok = pcall(function()
      if entry.Instance and entry.Instance.Parent then
        -- Tween color properties; other properties set instantly.
        local propType = typeof(entry.Instance[entry.Property])
        if propType == "Color3" then
          TweenService:Create(entry.Instance, tweenInfo, {
            [entry.Property] = Theme[entry.Key]
          }):Play()
        else
          entry.Instance[entry.Property] = Theme[entry.Key]
        end
      end
    end)
    if not ok then
      -- Element was destroyed; silently skip.
    end
  end
end

-- Returns a read-only copy of all available preset theme names.
function MIXUSFIELD_Library:GetThemeNames()
  local names = {}
  for k in pairs(THEMES) do
    table.insert(names, k)
  end
  return names
end

-- Registers a custom theme so it can be used via :SetTheme("MyTheme").
-- @param name   string identifier
-- @param colors { Accent, Background, Surface, Text, SubText, Stroke }
function MIXUSFIELD_Library:RegisterTheme(name, colors)
  THEMES[name] = colors
end

-- ============================================================
-- /// KEY SYSTEM
-- Blocks the UI until the correct key is entered.
-- Config keys:
--   Key       (string)  – the secret passphrase
--   Title     (string)  – window title shown to the user
--   SubTitle  (string)  – small subtitle/game name
--   Note      (string)  – hint or extra info text
--   Callback  (fn)      – fired when the correct key is accepted
-- ============================================================

-- Shows the key-entry screen. Returns a Funcs table with :Close().
-- The real UI (passed via `onSuccess`) is shown only after the correct key.
function MIXUSFIELD_Library:KeySystem(Config, onSuccess)
  local ValidKey  = Config.Key      or ""
  local Title     = Config.Title    or "MIXUSFIELD"
  local SubTitle  = Config.SubTitle or "Key Required"
  local Note      = Config.Note     or "Enter the key to access this script."
  local Callback  = Config.Callback or function() end

  -- Build a dedicated ScreenGui for the key system, separate from the main UI.
  local KeyGui = Custom:Create("ScreenGui", {
    ZIndexBehavior = Enum.ZIndexBehavior.Sibling,
  }, GetGuiParent())

  -- Full-screen dark overlay so nothing behind is clickable.
  local Overlay = Custom:Create("Frame", {
    Size                = UDim2.new(1, 0, 1, 0),
    BackgroundColor3    = Color3.fromRGB(0, 0, 0),
    BackgroundTransparency = 0.4,
    BorderSizePixel     = 0,
    Name                = "Overlay",
  }, KeyGui)

  -- Central card that holds all key-system controls.
  local Card = Custom:Create("Frame", {
    AnchorPoint         = Vector2.new(0.5, 0.5),
    Position            = UDim2.new(0.5, 0, 0.5, 0),
    Size                = UDim2.new(0, 360, 0, 200),
    BackgroundColor3    = Theme.Background,
    BackgroundTransparency = 0,
    BorderSizePixel     = 0,
    Name                = "KeyCard",
  }, Overlay)
  ThemeBind(Card, "BackgroundColor3", "Background")

  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 10) }, Card)

  -- Accent-colored outline on the card.
  local CardStroke = Custom:Create("UIStroke", {
    Color     = Theme.Accent,
    Thickness = 1.8,
  }, Card)
  ThemeBind(CardStroke, "Color", "Accent")

  -- Title label at the top of the card.
  local KeyTitle = Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = Title,
    TextColor3          = Theme.Text,
    TextSize            = 16,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 12),
    Size                = UDim2.new(1, -28, 0, 18),
    Name                = "KeyTitle",
  }, Card)
  ThemeBind(KeyTitle, "TextColor3", "Text")

  -- Colored subtitle / game name below the title.
  local KeySub = Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = SubTitle,
    TextColor3          = Theme.Accent,
    TextSize            = 13,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 32),
    Size                = UDim2.new(1, -28, 0, 14),
    Name                = "KeySub",
  }, Card)
  ThemeBind(KeySub, "TextColor3", "Accent")

  -- Thin divider line beneath the header area.
  local Divider = Custom:Create("Frame", {
    BackgroundColor3    = Theme.Stroke,
    BackgroundTransparency = 0,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 0, 0, 52),
    Size                = UDim2.new(1, 0, 0, 1),
    Name                = "Divider",
  }, Card)
  ThemeBind(Divider, "BackgroundColor3", "Stroke")

  -- Note / hint text in the middle of the card.
  local KeyNote = Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = Note,
    TextColor3          = Theme.SubText,
    TextSize            = 12,
    TextWrapped         = true,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 60),
    Size                = UDim2.new(1, -28, 0, 30),
    Name                = "KeyNote",
  }, Card)
  ThemeBind(KeyNote, "TextColor3", "SubText")

  -- Input field background frame.
  local InputFrame = Custom:Create("Frame", {
    BackgroundColor3    = Theme.Surface,
    BackgroundTransparency = 0,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 100),
    Size                = UDim2.new(1, -100, 0, 30),
    Name                = "KeyInputFrame",
  }, Card)
  ThemeBind(InputFrame, "BackgroundColor3", "Surface")
  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 6) }, InputFrame)

  -- Stroke on the input box to make it stand out.
  local InputStroke = Custom:Create("UIStroke", {
    Color       = Theme.Stroke,
    Thickness   = 1.2,
  }, InputFrame)
  ThemeBind(InputStroke, "Color", "Stroke")

  -- The actual TextBox the user types into.
  local KeyInput = Custom:Create("TextBox", {
    Font                = Enum.Font.GothamBold,
    PlaceholderText     = "Enter key here...",
    PlaceholderColor3   = Theme.SubText,
    Text                = "",
    TextColor3          = Theme.Text,
    TextSize            = 13,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 8, 0, 0),
    Size                = UDim2.new(1, -16, 1, 0),
    Name                = "KeyInput",
    ClearTextOnFocus    = false,
  }, InputFrame)
  ThemeBind(KeyInput, "TextColor3", "Text")
  ThemeBind(KeyInput, "PlaceholderColor3", "SubText")

  -- "Submit" button to the right of the input.
  local SubmitBtn = Custom:Create("TextButton", {
    Font                = Enum.Font.GothamBold,
    Text                = "Submit",
    TextColor3          = Theme.Text,
    TextSize            = 13,
    BackgroundColor3    = Theme.Accent,
    BackgroundTransparency = 0,
    BorderSizePixel     = 0,
    AnchorPoint         = Vector2.new(1, 0),
    Position            = UDim2.new(1, -14, 0, 100),
    Size                = UDim2.new(0, 72, 0, 30),
    Name                = "SubmitBtn",
  }, Card)
  ThemeBind(SubmitBtn, "BackgroundColor3", "Accent")
  ThemeBind(SubmitBtn, "TextColor3", "Text")
  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 6) }, SubmitBtn)

  -- Status label shown below the input (wrong key / loading feedback).
  local StatusLabel = Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = "",
    TextColor3          = Color3.fromRGB(220, 60, 60),  -- error red; not theme-bound by design
    TextSize            = 12,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 142),
    Size                = UDim2.new(1, -28, 0, 14),
    Name                = "StatusLabel",
  }, Card)

  -- Entrance animation: card slides up from slightly below center.
  Card.Position = UDim2.new(0.5, 0, 0.6, 0)
  TweenService:Create(Card, TweenInfo.new(0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
    Position = UDim2.new(0.5, 0, 0.5, 0)
  }):Play()

  local KeyFuncs = {}

  -- Destroys the key screen (called internally on success or externally to cancel).
  function KeyFuncs:Close()
    TweenService:Create(Card, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
      BackgroundTransparency = 1,
      Position = UDim2.new(0.5, 0, 0.4, 0),
    }):Play()
    task.wait(0.3)
    KeyGui:Destroy()
  end

  -- Validates the entered key and either accepts or rejects it.
  local function TryKey()
    local entered = KeyInput.Text
    if entered == ValidKey then
      -- Correct key: notify the user, fire the callback, dismiss the screen.
      StatusLabel.TextColor3 = Color3.fromRGB(60, 220, 100)
      StatusLabel.Text       = "✓ Accepted! Loading..."

      TweenService:Create(InputStroke, TweenInfo.new(0.2), {
        Color = Color3.fromRGB(60, 220, 100)
      }):Play()

      task.wait(0.8)
      Callback()
      KeyFuncs:Close()

      -- Signal the caller to proceed with building the real UI.
      if type(onSuccess) == "function" then
        onSuccess()
      end
    else
      -- Wrong key: flash the input stroke red and show an error message.
      StatusLabel.TextColor3 = Color3.fromRGB(220, 60, 60)
      StatusLabel.Text       = "✗ Incorrect key. Try again."

      TweenService:Create(InputStroke, TweenInfo.new(0.1), {
        Color = Color3.fromRGB(220, 60, 60)
      }):Play()

      task.wait(0.5)

      -- Reset the stroke color back to the theme default.
      TweenService:Create(InputStroke, TweenInfo.new(0.2), {
        Color = Theme.Stroke
      }):Play()
    end
  end

  -- Wire the submit button and Enter key to the validator.
  SubmitBtn.Activated:Connect(function()
    CircleClick(SubmitBtn, Player:GetMouse().X, Player:GetMouse().Y)
    TryKey()
  end)

  KeyInput.FocusLost:Connect(function(enterPressed)
    if enterPressed then TryKey() end
  end)

  return KeyFuncs
end

-- ============================================================
-- /// NOTIFICATION SYSTEM
-- Self-contained floating toast notifications.
-- Config: Title, Description, Content, Time (tween), Delay (auto-close)
-- ============================================================
function MIXUSFIELD_Library:SetNotification(Config)
  local Title       = Config[1] or Config.Title       or ""
  local Description = Config[2] or Config.Description or ""
  local Content     = Config[3] or Config.Content     or ""
  local Time        = tonumber(Config[5] or Config.Time  or 0.5)
  local Delay       = tonumber(Config[6] or Config.Delay or 5)

  local NotifGui = Custom:Create("ScreenGui", {
    ZIndexBehavior = Enum.ZIndexBehavior.Sibling,
  }, GetGuiParent())

  -- Anchor frame in the bottom-right corner that stacks notifications.
  local NotifLayout = Custom:Create("Frame", {
    AnchorPoint         = Vector2.new(1, 1),
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(1, -30, 1, -30),
    Size                = UDim2.new(0, 320, 1, 0),
    Name                = "NotifLayout",
  }, NotifGui)

  local Count = 0

  -- Re-stack remaining notifications when one is removed.
  NotifLayout.ChildRemoved:Connect(function()
    Count = 0
    local info = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut)
    for _, v in ipairs(NotifLayout:GetChildren()) do
      local newPos = UDim2.new(0, 0, 1, -((v.Size.Y.Offset + 12) * Count))
      TweenService:Create(v, info, { Position = newPos }):Play()
      Count = Count + 1
    end
  end)

  -- Calculate vertical offset so the new toast appears above existing ones.
  local _Count = 0
  for _, v in ipairs(NotifLayout:GetChildren()) do
    _Count = -(v.Position.Y.Offset) + v.Size.Y.Offset + 12
  end

  local NotifFrame = Custom:Create("Frame", {
    AnchorPoint         = Vector2.new(0, 1),
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 0, 1, -_Count),
    Size                = UDim2.new(1, 0, 0, 150),
    Name                = "NotifFrame",
  }, NotifLayout)

  -- Actual colored card inside the animation wrapper.
  local NotifCard = Custom:Create("Frame", {
    BackgroundColor3    = Theme.Background,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 400, 0, 0),   -- starts off-screen right
    Size                = UDim2.new(1, 0, 1, 0),
    Name                = "NotifCard",
  }, NotifFrame)
  ThemeBind(NotifCard, "BackgroundColor3", "Background")

  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 8) }, NotifCard)

  -- Accent-colored left border strip for visual identity.
  local AccentBar = Custom:Create("Frame", {
    BackgroundColor3    = Theme.Accent,
    BorderSizePixel     = 0,
    Size                = UDim2.new(0, 3, 1, 0),
    Name                = "AccentBar",
  }, NotifCard)
  ThemeBind(AccentBar, "BackgroundColor3", "Accent")
  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 8) }, AccentBar)

  -- UIStroke around the whole notification card.
  local NotifStroke = Custom:Create("UIStroke", {
    Color     = Theme.Stroke,
    Thickness = 1.2,
  }, NotifCard)
  ThemeBind(NotifStroke, "Color", "Stroke")

  -- Header bar with title and dismiss button.
  local Top = Custom:Create("Frame", {
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Size                = UDim2.new(1, 0, 0, 36),
    Name                = "Top",
  }, NotifCard)

  Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = Title,
    TextColor3          = Theme.Text,
    TextSize            = 14,
    TextXAlignment      = Enum.TextXAlignment.Left,
    BackgroundTransparency = 1,
    BorderSizePixel     = 0,
    Position            = UDim2.new(0, 14, 0, 0),
    Size                = UDim2.new(1, -50, 1, 0),
    Name                = "NotifTitle",
  }, Top)

  Custom:Create("TextLabel", {
    Font                = Enum.Font.GothamBold,
    Text                = Description,
    TextColor3          = Theme.Accent,
    TextSize            = 13,

    NotifFrame.Size = UDim2.new(1, 0, 0, BodyLabel.AbsoluteSize.Y + 48)
  end

  local dismissed = false

  -- Slides the notification back off-screen and removes it.
  function Notification:Close()
    if dismissed then return end
    dismissed = true
    TweenService:Create(NotifCard, TweenInfo.new(Time, Enum.EasingStyle.Back, Enum.EasingDirection.InOut), {
      Position = UDim2.new(0, 400, 0, 0)
    }):Play()
    task.wait(Time / 1.2)
    NotifFrame:Destroy()
    dismissed = false
  end

  CloseBtn.Activated:Connect(function() Notification:Close() end)

  -- Slide in from the right.
  TweenService:Create(NotifCard, TweenInfo.new(Time, Enum.EasingStyle.Back, Enum.EasingDirection.InOut), {
    Position = UDim2.new(0, 0, 0, 0)
  }):Play()

  -- Auto-dismiss after `Delay` seconds.
  task.delay(Delay, function() Notification:Close() end)

  return Notification

      -- Recalculates the outer scroll canvas height.
      local function UpdateSizeScroll()
        local offset = 0
        for _, child in pairs(ScrolLayers:GetChildren()) do
          if child.Name ~= "UIListLayout" then
            offset = offset + 3 + child.Size.Y.Offset
          end
        end
        ScrolLayers.CanvasSize = UDim2.new(0, 0, 0, offset)
      end

      -- Recalculates and tweens the section to its correct open height.
      local function UpdateSizeSection()
        if OpenSection then
          local height = 38
          for _, v in pairs(SectionAdd:GetChildren()) do
            if v.Name ~= "UIListLayout" and v.Name ~= "UICorner" then
              height = height + v.Size.Y.Offset + 3
            end
          end
          TweenService:Create(FeatureFrame, TweenInfo.new(0.1), { Rotation = 90 }):Play()
          TweenService:Create(Section, TweenInfo.new(0.1), { Size = UDim2.new(1, 1, 0, height) }):Play()
          TweenService:Create(SectionAdd, TweenInfo.new(0.1), { Size = UDim2.new(1, 0, 0, height - 38) }):Play()
          TweenService:Create(SectionDecide, TweenInfo.new(0.1), { Size = UDim2.new(1, 0, 0, 2) }):Play()
          task.wait(0.5)
          UpdateSizeScroll()
        end
      end

          TweenService:Create(Section, TweenInfo.new(0.1), { Size = UDim2.new(1, 1, 0, 30) }):Play()
          TweenService:Create(SectionDecide, TweenInfo.new(0.1), { Size = UDim2.new(0, 0, 0, 2) }):Play()
          OpenSection = false
          task.wait(0.1)
          UpdateSizeScroll()
        else
          OpenSection = true
          UpdateSizeSection()
        end
      end

        ThemeBind(SliderFill, "BackgroundColor3", "Accent")

        Custom:Create("UICorner", {}, SliderFill)

        -- The draggable handle circle at the end of the fill.
        local SliderHandle = Custom:Create("Frame", {
          AnchorPoint         = Vector2.new(1, 0.5),
          BackgroundColor3    = Theme.Accent,
          BorderSizePixel     = 0,
          Position            = UDim2.new(1, 4, 0.5, 0),
          Size                = UDim2.new(0, 10, 0, 10),
          Name                = "SliderHandle",
        }, SliderFill)
        ThemeBind(SliderHandle, "BackgroundColor3", "Accent")

        Custom:Create("UICorner", {}, SliderHandle)

        -- Handle outline stroke.
        local HandleStroke = Custom:Create("UIStroke", {
          Color       = Theme.Text,
          Thickness   = 1.5,
        }, SliderHandle)
        ThemeBind(HandleStroke, "Color", "Text")

        local Dragging = false

        -- Rounds a number to the nearest increment.
        local function Round(n, factor)
          local result = math.floor(n / factor + (math.sign(n) * 0.5)) * factor
          if result < 0 then result = result + factor end
          return result
        end

        function SFuncs:Set(value)
          value         = math.clamp(Round(value, Increment), SMin, SMax)
          SFuncs.Value  = value
          TextBox.Text  = tostring(value)
          TweenService:Create(SliderFill, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.fromScale((value - SMin) / (SMax - SMin), 1)
          }):Play()
        end

        -- Mouse/touch drag on the track.
        SliderTrack.InputBegan:Connect(function(input)
          if input.UserInputType == Enum.UserInputType.MouseButton1
          or input.UserInputType == Enum.UserInputType.Touch then
            Dragging = true
          end
        end)

        SliderTrack.InputEnded:Connect(function(input)
          if input.UserInputType == Enum.UserInputType.MouseButton1
          or input.UserInputType == Enum.UserInputType.Touch then
            Dragging = false
            SCallback(SFuncs.Value)
          end
        end)

        local _LastX = nil
        UserInputService.InputChanged:Connect(function(input)
          if Dragging then
            local cx = input.Position.X
            if cx ~= _LastX then
              _LastX = cx
              local scale = math.clamp((cx - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1)
              SFuncs:Set(SMin + ((SMax - SMin) * scale))
            end
          end
        end)

        -- Validate typed number input (strip non-digits, clamp to max).
        TextBox:GetPropertyChangedSignal("Text"):Connect(function()
          local valid = TextBox.Text:gsub("[^%d]", "")
          if valid ~= "" then
            TextBox.Text = tostring(math.min(tonumber(valid), SMax))
          else
            TextBox.Text = "0"
          end
        end)

        TextBox.FocusLost:Connect(function()
          local val = tonumber(TextBox.Text) or 0
          SFuncs:Set(val)
          SCallback(SFuncs.Value)
        end)

        SFuncs:Set(SDefault)
        SCallback(SFuncs.Value)

        ItemCount += 1
        return SFuncs
      end

              if drop:FindFirstChild("ChooseFrame") then
                TweenService:Create(drop.ChooseFrame, info, { Size = size }):Play()

                  TweenService:Create(drop.ChooseFrame.UIStroke, info, { Transparency = sAlpha }):Play()
                end
              end
              TweenService:Create(drop, info, { BackgroundTransparency = alpha }):Play()
            end
          end

          local joined = table.concat(DFuncs.Value, ", ")
          OptionSelecting.Text = joined ~= "" and joined or "Select Options"
          DCallback(DFuncs.Value)

              end
            end
            ScrollSelect.CanvasSize = UDim2.new(0, 0, 0, h)
          end

          DropCount += 1
        end

        -- Replaces the entire option list and applies new selection.
        function DFuncs:Refresh(list, selecting)

          selecting = selecting or {}
          DFuncs:Clear()
          for _, name in ipairs(list) do
            DFuncs:AddOption(name)
          end
          DFuncs.Options = list
          DFuncs:Set(selecting)
        end

        DFuncs:Refresh(DFuncs.Options, DFuncs.Value)

        ItemCount  += 1
        CountDropdown += 1
        return DFuncs
      end -- AddDropdown

      ItemCount += 1
      return Item
    end -- AddSection

    CountSection += 1
    CountTab    += 1
    return Sections
  end -- CreateTab

  return Tabs
