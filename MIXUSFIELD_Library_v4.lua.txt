-- /// SAVE SYSTEM
-- ============================================================
local SaveSystem = {}
SaveSystem._data = {}
SaveSystem._file = "MixusfieldSave_v4.json"

function SaveSystem:_load()
  local ok, raw = pcall(readfile, self._file)
  if ok and raw and #raw > 0 then
    local jok, parsed = pcall(HttpService.JSONDecode, HttpService, raw)
    if jok and type(parsed) == "table" then self._data = parsed end
  end
end

function SaveSystem:_flush()
  local ok, encoded = pcall(HttpService.JSONEncode, HttpService, self._data)
  if ok then pcall(writefile, self._file, encoded) end
end

function SaveSystem:Get(key, default)
  return self._data[key] ~= nil and self._data[key] or default
end

function SaveSystem:Set(key, value)
  self._data[key] = value
  self:_flush()
end

function SaveSystem:Clear(key)
  if key then self._data[key] = nil
  else        self._data = {} end
  self:_flush()
end

function SaveSystem:SetFile(filename)
  self._file = filename
  self:_load()
end

function SaveSystem:Export()
  local ok, enc = pcall(HttpService.JSONEncode, HttpService, self._data)
  return ok and enc or "{}"
end

function SaveSystem:Import(json)
  local ok, parsed = pcall(HttpService.JSONDecode, HttpService, json)
  if ok and type(parsed) == "table" then
    for k, v in pairs(parsed) do self._data[k] = v end
    self:_flush()
    return true
  end
  return false
end

SaveSystem:_load()

local ThemeListeners = {}

local function ThemeBind(inst, prop, key)
  if not inst then return end
  table.insert(ThemeListeners, { Instance=inst, Property=prop, Key=key })
  pcall(function()
    if inst and inst.Parent then inst[prop] = Theme[key] end
  end)
end

local Custom = {}

function Custom:Create(Name, Properties, Parent)
  local inst = Instance.new(Name)
  for i, v in pairs(Properties) do pcall(function() inst[i] = v end) end
  if Parent then inst.Parent = Parent end
  return inst
end

local function CircleClick(Button, X, Y)
  task.spawn(function()
    if not Button or not Button.Parent then return end
    Button.ClipsDescendants = true
    local Circle = Instance.new("ImageLabel")
    Circle.Image             = "rbxassetid://106471194043211"
    Circle.ImageColor3       = Color3.fromRGB(255,255,255)
    Circle.ImageTransparency = 0.88
    Circle.BackgroundTransparency = 1
    Circle.ZIndex  = 10
    Circle.Name    = "Ripple"
    Circle.Parent  = Button

    Circle.Position = UDim2.new(0, X - Button.AbsolutePosition.X, 0, Y - Button.AbsolutePosition.Y)
    local Size = math.max(Button.AbsoluteSize.X, Button.AbsoluteSize.Y) * 1.6
    local info = TweenInfo.new(0.45, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    local t = TweenService:Create(Circle, info, {
      Size = UDim2.new(0,Size,0,Size),
      Position = UDim2.new(0.5,-Size/2,0.5,-Size/2),
      ImageTransparency = 1,
    })
    t:Play()
    t.Completed:Connect(function() pcall(function() Circle:Destroy() end) end)
  end)
end

-- /// DRAGGABLE HELPER
-- ============================================================
local function MakeDraggable(topbar, object)
  local dragging, dragStart, startPos = false, nil, nil
  topbar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
      dragging = true; dragStart = input.Position; startPos = object.Position
      input.Changed:Connect(function()
        if input.UserInputState == Enum.UserInputState.End then dragging = false end
      end)
    end
  end)
  topbar.InputChanged:Connect(function(input)
    if dragging and (
      input.UserInputType == Enum.UserInputType.MouseMovement
      or input.UserInputType == Enum.UserInputType.Touch
    ) then
      local d = input.Position - dragStart
      object.Position = UDim2.new(
        startPos.X.Scale, startPos.X.Offset + d.X,
        startPos.Y.Scale, startPos.Y.Offset + d.Y)
    end
  end)
end

-- ============================================================
-- /// THEME PUBLIC API
-- ============================================================
function MIXUSFIELD_Library:SetTheme(nameOrTable)
  local newTheme
  if type(nameOrTable) == "string" then
    for k, v in pairs(THEMES) do
      if k:lower() == nameOrTable:lower() then newTheme = v; break end
    end
    if not newTheme then warn("[MIXUSFIELD] Unknown theme: "..tostring(nameOrTable)); return end
  elseif type(nameOrTable) == "table" then
    newTheme = nameOrTable
  else
    warn("[MIXUSFIELD] SetTheme expects a string or color table."); return
  end

  for k, v in pairs(newTheme) do Theme[k] = v end

  local tw = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
  for _, entry in ipairs(ThemeListeners) do
    pcall(function()
      if entry.Instance and entry.Instance.Parent then
        if typeof(entry.Instance[entry.Property]) == "Color3" then
          TweenService:Create(entry.Instance, tw, {[entry.Property]=Theme[entry.Key]}):Play()
        else
          entry.Instance[entry.Property] = Theme[entry.Key]
        end
      end
    end)
  end
end

function MIXUSFIELD_Library:GetThemeNames()
  local names = {}
  for k in pairs(THEMES) do table.insert(names, k) end
  table.sort(names)
  return names
end

function MIXUSFIELD_Library:RegisterTheme(name, colors)
  THEMES[name] = colors
end

-- ============================================================
-- /// KEY SYSTEM
-- ============================================================
function MIXUSFIELD_Library:KeySystem(Config, onSuccess)
  local ValidKey = Config.Key      or ""
  local Title    = Config.Title    or "MIXUSFIELD"
  local SubTitle = Config.SubTitle or "Key Required"
  local Note     = Config.Note     or "Enter the key below to continue."
  local Callback = Config.Callback or function() end
  local SKey     = Config.SaveKey
  local Expiry   = tonumber(Config.Expiry) or 0

  if SKey then
    local saved   = SaveSystem:Get(SKey.."_key")
    local savedAt = SaveSystem:Get(SKey.."_savedAt", 0)
    local valid   = saved == ValidKey
    if valid and Expiry > 0 then valid = (os.time()-savedAt) < Expiry end
    if valid then
      Callback()
      if type(onSuccess)=="function" then onSuccess() end
      return {Close=function() end}
    end
  end

  local KeyGui = Custom:Create("ScreenGui",{
    ZIndexBehavior=Enum.ZIndexBehavior.Sibling, Name="MixusKeyGui",
  }, GetGuiParent())

  local Overlay = Custom:Create("Frame",{
    Size=UDim2.new(1,0,1,0), BackgroundColor3=Color3.fromRGB(0,0,0),
    BackgroundTransparency=0.35, BorderSizePixel=0,
  }, KeyGui)

  local Card = Custom:Create("Frame",{
    AnchorPoint=Vector2.new(0.5,0.5), Position=UDim2.new(0.5,0,0.6,0),
    Size=UDim2.new(0,380,0,220), BackgroundColor3=Theme.Background,
    BorderSizePixel=0, Name="KeyCard",
  }, Overlay)
  ThemeBind(Card,"BackgroundColor3","Background")
  Custom:Create("UICorner",{CornerRadius=UDim.new(0,12)}, Card)
  local CardStroke = Custom:Create("UIStroke",{Color=Theme.Accent,Thickness=1.8}, Card)
  ThemeBind(CardStroke,"Color","Accent")

  local TopStrip = Custom:Create("Frame",{
    BackgroundColor3=Theme.Accent, BackgroundTransparency=0.85,
    BorderSizePixel=0, Size=UDim2.new(1,0,0,42),
  }, Card)
  ThemeBind(TopStrip,"BackgroundColor3","Accent")
  Custom:Create("UICorner",{CornerRadius=UDim.new(0,12)}, TopStrip)
  Custom:Create("Frame",{
    BackgroundColor3=Theme.Accent, BackgroundTransparency=0.85, BorderSizePixel=0,
    Position=UDim2.new(0,0,0,20), Size=UDim2.new(1,0,0,22),
  }, Card)

  local function KL(txt,ts,tc,py,bold)
    Custom:Create("TextLabel",{
      Font=bold and Enum.Font.GothamBold or Enum.Font.Gotham,
      Text=txt, TextColor3=tc, TextSize=ts,
      TextXAlignment=Enum.TextXAlignment.Left, BackgroundTransparency=1, BorderSizePixel=0,
      Position=UDim2.new(0,16,0,py), Size=UDim2.new(1,-32,0,ts+3),
    }, Card)
  end
  KL(Title,15,Theme.Text,10,true)
  KL(SubTitle,12,Theme.Accent,30,true)
  Custom:Create("Frame",{
    BackgroundColor3=Theme.Stroke, BackgroundTransparency=0.3, BorderSizePixel=0,
    Position=UDim2.new(0,0,0,48), Size=UDim2.new(1,0,0,1),
  }, Card)
  KL(Note,12,Theme.SubText,58,false)

  local InputFrame = Custom:Create("Frame",{
    BackgroundColor3=Theme.Surface, BorderSizePixel=0,
    Position=UDim2.new(0,16,0,98), Size=UDim2.new(1,-104,0,32),
  }, Card)
  ThemeBind(InputFrame,"BackgroundColor3","Surface")
  Custom:Create("UICorner",{CornerRadius=UDim.new(0,7)}, InputFrame)
  local InputStroke = Custom:Create("UIStroke",{Color=Theme.Stroke,Thickness=1.3}, InputFrame)
  ThemeBind(InputStroke,"Color","Stroke")

  local KeyInput = Custom:Create("TextBox",{
    Font=Enum.Font.GothamBold, PlaceholderText="Enter key...",
    PlaceholderColor3=Theme.SubText, Text="", TextColor3=Theme.Text, TextSize=13,
    TextXAlignment=Enum.TextXAlignment.Left, BackgroundTransparency=1, BorderSizePixel=0,
    Position=UDim2.new(0,10,0,0), Size=UDim2.new(1,-16,1,0), ClearTextOnFocus=false,
  }, InputFrame)
  ThemeBind(KeyInput,"TextColor3","Text")
  ThemeBind(KeyInput,"PlaceholderColor3","SubText")

  local SubmitBtn = Custom:Create("TextButton",{
    Font=Enum.Font.GothamBold, Text="Submit", TextColor3=Theme.Text, TextSize=13,
    BackgroundColor3=Theme.Accent, BorderSizePixel=0,
    AnchorPoint=Vector2.new(1,0), Position=UDim2.new(1,-16,0,98),
    Size=UDim2.new(0,76,0,32),
  }, Card)
  ThemeBind(SubmitBtn,"BackgroundColor3","Accent")
  ThemeBind(SubmitBtn,"TextColor3","Text")
  Custom:Create("UICorner",{CornerRadius=UDim.new(0,7)}, SubmitBtn)

  local StatusLabel = Custom:Create("TextLabel",{
    Font=Enum.Font.GothamBold, Text="", TextColor3=Color3.fromRGB(220,60,60), TextSize=12,
    TextXAlignment=Enum.TextXAlignment.Left, BackgroundTransparency=1, BorderSizePixel=0,
    Position=UDim2.new(0,16,0,140), Size=UDim2.new(1,-32,0,14),
  }, Card)
  KL(Expiry>0 and ("Key expires "..math.floor(Expiry/3600).."h after first use") or "", 11, Theme.SubText, 158, false)

  TweenService:Create(Card, TweenInfo.new(0.4,Enum.EasingStyle.Back,Enum.EasingDirection.Out),
    {Position=UDim2.new(0.5,0,0.5,0)}):Play()

  local KeyFuncs = {}
  function KeyFuncs:Close()
    TweenService:Create(Card, TweenInfo.new(0.25,Enum.EasingStyle.Quad,Enum.EasingDirection.In),
      {Position=UDim2.new(0.5,0,0.4,0),BackgroundTransparency=1}):Play()
    task.wait(0.3); KeyGui:Destroy()
  end

  local function TryKey()
    local entered = KeyInput.Text
    if entered == ValidKey then
      StatusLabel.TextColor3=Color3.fromRGB(60,210,100)
      StatusLabel.Text="✓ Accepted! Loading..."
      TweenService:Create(InputStroke,TweenInfo.new(0.2),{Color=Color3.fromRGB(60,210,100)}):Play()
      if SKey then
        SaveSystem:Set(SKey.."_key", entered)
        SaveSystem:Set(SKey.."_savedAt", os.time())
      end
      task.wait(0.8); Callback(); KeyFuncs:Close()
      if type(onSuccess)=="function" then onSuccess() end
    else
      StatusLabel.TextColor3=Color3.fromRGB(220,60,60)
      StatusLabel.Text="✗ Incorrect key."
      TweenService:Create(InputStroke,TweenInfo.new(0.1),{Color=Color3.fromRGB(220,60,60)}):Play()
      task.wait(0.5)
      TweenService:Create(InputStroke,TweenInfo.new(0.2),{Color=Theme.Stroke}):Play()
      StatusLabel.Text=""
    end
  end

  SubmitBtn.Activated:Connect(function()
    CircleClick(SubmitBtn, Player:GetMouse().X, Player:GetMouse().Y); TryKey()
  end)
  KeyInput.FocusLost:Connect(function(enter) if enter then TryKey() end end)
  return KeyFuncs
end

        KDisp.Activated:Connect(function()

            if lConn then lConn:Disconnect(); lConn=nil end
            listening=false; KDisp.Text=tostring(KFuncs.Value.Name)
            TweenService:Create(stroke,TweenInfo.new(0.1),{Color=Theme.Stroke,Transparency=0.5}):Play()
            return
          end
          listening=true; KDisp.Text="[ ... ]"
          TweenService:Create(stroke,TweenInfo.new(0.1),{Color=Theme.Accent,Transparency=0.15}):Play()
          lConn=UserInputService.InputBegan:Connect(function(inp,gp)
            if gp then return end
            if inp.UserInputType==Enum.UserInputType.Keyboard then
              if inp.KeyCode==Enum.KeyCode.Escape then
                listening=false; lConn:Disconnect(); lConn=nil
                KDisp.Text=tostring(KFuncs.Value.Name)
                TweenService:Create(stroke,TweenInfo.new(0.1),{Color=Theme.Stroke,Transparency=0.5}):Play()
                return
              end
              KFuncs:Set(inp.KeyCode); lConn:Disconnect(); lConn=nil; listening=false
              TweenService:Create(stroke,TweenInfo.new(0.1),{Color=Theme.Stroke,Transparency=0.5}):Play()
            end
          end)
        end)
        UserInputService.InputBegan:Connect(function(inp,gp)
          if not gp and not listening and inp.KeyCode==KFuncs.Value then
            task.spawn(KCallback,KFuncs.Value)
          end
        end)
        function KFuncs:Set(kc)
          KFuncs.Value=kc; KDisp.Text=kc.Name
          if KSaveKey then SaveSystem:Set(KSaveKey,kc.Name) end
          task.spawn(KCallback,kc)

      end

              cS=math.clamp((inp.Position.X-SVSq.AbsolutePosition.X)/SVSq.AbsoluteSize.X,0,1)
              cV=1-math.clamp((inp.Position.Y-SVSq.AbsolutePosition.Y)/SVSq.AbsoluteSize.Y,0,1)
              SVCur.Position=UDim2.new(cS,0,1-cV,0); Apply()
            end
          end)

          local hueDrag=false
          HBar.InputBegan:Connect(function(inp)
            if inp.UserInputType==Enum.UserInputType.MouseButton1
            or inp.UserInputType==Enum.UserInputType.Touch then
              hueDrag=true
              cH=1-math.clamp((inp.Position.Y-HBar.AbsolutePosition.Y)/HBar.AbsoluteSize.Y,0,1)
              HCur.Position=UDim2.new(0.5,0,1-cH,0); Apply()
            end
          end)
          HBar.InputEnded:Connect(function(inp)
            if inp.UserInputType==Enum.UserInputType.MouseButton1
            or inp.UserInputType==Enum.UserInputType.Touch then hueDrag=false end
          end)
          UserInputService.InputChanged:Connect(function(inp)
            if hueDrag and (inp.UserInputType==Enum.UserInputType.MouseMovement
            or inp.UserInputType==Enum.UserInputType.Touch) then
              cH=1-math.clamp((inp.Position.Y-HBar.AbsolutePosition.Y)/HBar.AbsoluteSize.Y,0,1)
              HCur.Position=UDim2.new(0.5,0,1-cH,0); Apply()
            end
          end)

          HexIn.FocusLost:Connect(function()
            local c=HexToC(HexIn.Text)
            if c then
              cH,cS,cV=Color3.toHSV(c)
              SVCur.Position=UDim2.new(cS,0,1-cV,0)
              HCur.Position=UDim2.new(0.5,0,1-cH,0); Apply()
            end
          end)
          Apply(); task.defer(UpdateSizeSection)
        end
        RowBtn.Activated:Connect(function()
          CircleClick(RowBtn,Player:GetMouse().X,Player:GetMouse().Y)
          if not expanded then
            expanded=true; BuildPicker()
            TweenService:Create(stroke,TweenInfo.new(0.15),{Color=Theme.Accent,Transparency=0.2}):Play()
          else
            expanded=false
            TweenService:Create(stroke,TweenInfo.new(0.15),{Color=Theme.Stroke,Transparency=0.5}):Play()
            if PickerFrame then PickerFrame:Destroy(); PickerFrame=nil end
            task.defer(UpdateSizeSection)
          end
        end)
        function CPFuncs:Set(color)
          CPFuncs.Value=color; Swatch.BackgroundColor3=color
          cH,cS,cV=Color3.toHSV(color)
          if CPSaveKey then SaveSystem:Set(CPSaveKey,{r=color.R,g=color.G,b=color.B}) end
          task.spawn(CPCallback,color)
        end
        ItemCount+=1; return CPFuncs
      end

      ----------- Dropdown -----------
      function Item:AddDropdown(Config)
        local Multi=Config[3] or Config.Multi or false
        local Options=Config[4] or Config.Options or {}
        local DDef=Config[5] or Config.Default or {}
        local DCallback=Config[6] or Config.Callback or function() end
        local DSaveKey=Config[7] or Config.SaveKey or nil
        local DFuncs={Value=DDef, Options=Options}
        if DSaveKey then
          local sv=SaveSystem:Get(DSaveKey)
          if type(sv)=="table" then DDef=sv
          elseif type(sv)=="string" then DDef={sv} end
          DFuncs.Value=DDef
        end
        local row,stroke=MakeRow(36,"Dropdown")
        MakeRowLabels(row,Config[1] or Config.Title or "",Config[2] or Config.Content or "",160)
        local SelectBox=Custom:Create("Frame",{
          AnchorPoint=Vector2.new(1,0.5), BackgroundColor3=Theme.SurfaceB, BackgroundTransparency=0.1,
          BorderSizePixel=0, Position=UDim2.new(1,-8,0.5,0), Size=UDim2.new(0,148,0,24),
          LayoutOrder=CountDropdown,
        }, row)
        ThemeBind(SelectBox,"BackgroundColor3","SurfaceB")
        Custom:Create("UICorner",{CornerRadius=UDim.new(0,6)}, SelectBox)
        local SelStroke=Custom:Create("UIStroke",{Color=Theme.Stroke,Thickness=1.2,Transparency=0.4}, SelectBox)
        ThemeBind(SelStroke,"Color","Stroke")
        local OptionLbl=Custom:Create("TextLabel",{
          Font=Enum.Font.GothamBold, Text="Select...", TextColor3=Theme.SubText, TextSize=11,
          TextXAlignment=Enum.TextXAlignment.Left, AnchorPoint=Vector2.new(0,0.5),
          BackgroundTransparency=1, BorderSizePixel=0,
          Position=UDim2.new(0,7,0.5,0), Size=UDim2.new(1,-28,1,-4),
        }, SelectBox)
        ThemeBind(OptionLbl,"TextColor3","SubText")
        Custom:Create("ImageLabel",{
          Image="rbxassetid://90200523188815", ImageColor3=Theme.SubText,
          AnchorPoint=Vector2.new(1,0.5), BackgroundTransparency=1, BorderSizePixel=0,
          Position=UDim2.new(1,1,0.5,0), Size=UDim2.new(0,20,0,20),
        }, SelectBox)
        local DropBtn=Custom:Create("TextButton",{
          Font=Enum.Font.SourceSans, Text="", BackgroundTransparency=0.999,
          BorderSizePixel=0, Size=UDim2.new(1,0,1,0),
        }, row)
        Custom:Create("Folder",{Name="DropFolder_"..tostring(CountDropdown)}, DropdownFolder)
        local ScrollSelect=Custom:Create("ScrollingFrame",{
          CanvasSize=UDim2.new(0,0,0,0), ScrollBarThickness=2,
          ScrollBarImageColor3=Theme.Accent, Active=true, LayoutOrder=CountDropdown,
          BackgroundTransparency=1, BorderSizePixel=0, Size=UDim2.new(1,0,1,0),
        }, DropdownFolder)
        ThemeBind(ScrollSelect,"ScrollBarImageColor3","Accent")
        Custom:Create("UIListLayout",{Padding=UDim.new(0,3),SortOrder=Enum.SortOrder.LayoutOrder}, ScrollSelect)
        Custom:Create("UIPadding",{PaddingTop=UDim.new(0,2)}, ScrollSelect)
        -- Search
        local SearchBar=Custom:Create("TextBox",{
          Font=Enum.Font.GothamBold, PlaceholderText="Search...", PlaceholderColor3=Theme.SubText,
          Text="", TextColor3=Theme.Text, TextSize=11, BackgroundColor3=Theme.SurfaceB,
          BackgroundTransparency=0.2, BorderSizePixel=0, Size=UDim2.new(1,0,0,24),
        }, ScrollSelect)
        ThemeBind(SearchBar,"TextColor3","Text"); ThemeBind(SearchBar,"PlaceholderColor3","SubText")
        ThemeBind(SearchBar,"BackgroundColor3","SurfaceB")
        Custom:Create("UICorner",{CornerRadius=UDim.new(0,4)}, SearchBar)
        local SBStroke=Custom:Create("UIStroke",{Color=Theme.Stroke,Thickness=1,Transparency=0.4}, SearchBar)
        ThemeBind(SBStroke,"Color","Stroke")
        SearchBar.Focused:Connect(function()
          TweenService:Create(SBStroke,TweenInfo.new(0.1),{Color=Theme.Accent,Transparency=0}):Play()
        end)
        SearchBar.FocusLost:Connect(function()
          TweenService:Create(SBStroke,TweenInfo.new(0.1),{Color=Theme.Stroke,Transparency=0.4}):Play()
        end)
        SearchBar:GetPropertyChangedSignal("Text"):Connect(function()
          local q=string.lower(SearchBar.Text)
          for _,v in pairs(ScrollSelect:GetChildren()) do
            if v:IsA("Frame") and v.Name=="DropOption" then
              local lb=v:FindFirstChild("OptionText")
              if lb then v.Visible=string.find(string.lower(lb.Text),q,1,true)~=nil end
            end
          end
        end)
        local DropCount=0
        function DFuncs:Clear()
          for _,f in pairs(ScrollSelect:GetChildren()) do
            if f.Name=="DropOption" then f:Destroy() end
          end
          DFuncs.Value={}; DFuncs.Options={}; OptionLbl.Text="Select..."
        end
        function DFuncs:Set(value)
          DFuncs.Value=value or DFuncs.Value
          local tw=TweenInfo.new(0.15,Enum.EasingStyle.Quad,Enum.EasingDirection.InOut)
          for _,drop in pairs(ScrollSelect:GetChildren()) do
            if drop.Name=="DropOption" then
              local lb=drop:FindFirstChild("OptionText")
              local found=lb and table.find(DFuncs.Value,lb.Text)
              TweenService:Create(drop,tw,{BackgroundTransparency=found and 0.7 or 0.96}):Play()
              local cb=drop:FindFirstChild("ChooseBar")
              if cb then TweenService:Create(cb,tw,{Size=found and UDim2.new(0,3,0,14) or UDim2.new(0,0,0,14)}):Play() end
            end
          end
          local joined=table.concat(DFuncs.Value,", ")
          OptionLbl.Text=joined~="" and joined or "Select..."
          if DSaveKey then SaveSystem:Set(DSaveKey,Multi and DFuncs.Value or (DFuncs.Value[1] or "")) end
          task.spawn(DCallback,DFuncs.Value)
        end
        function DFuncs:AddOption(optName)
          optName=optName or "Option"
          local opt=Custom:Create("Frame",{
            BackgroundColor3=Theme.Surface, BackgroundTransparency=0.96, BorderSizePixel=0,
            LayoutOrder=DropCount, Size=UDim2.new(1,0,0,28), Name="DropOption",
          }, ScrollSelect)
          ThemeBind(opt,"BackgroundColor3","Surface")
          Custom:Create("UICorner",{CornerRadius=UDim.new(0,4)}, opt)
          local os2=Custom:Create("UIStroke",{Color=Theme.Stroke,Thickness=1,Transparency=0.6}, opt)
          ThemeBind(os2,"Color","Stroke")
          local ChooseBar=Custom:Create("Frame",{
            AnchorPoint=Vector2.new(0,0.5), BackgroundColor3=Theme.Accent, BorderSizePixel=0,
            Position=UDim2.new(0,2,0.5,0), Size=UDim2.new(0,0,0,14), Name="ChooseBar",
          }, opt)
          ThemeBind(ChooseBar,"BackgroundColor3","Accent")
          Custom:Create("UICorner",{}, ChooseBar)
          local optLbl=Custom:Create("TextLabel",{
            Font=Enum.Font.GothamBold, Text=optName, TextColor3=Theme.Text, TextSize=11,
            TextXAlignment=Enum.TextXAlignment.Left, BackgroundTransparency=1, BorderSizePixel=0,
            Position=UDim2.new(0,10,0,0), Size=UDim2.new(1,-14,1,0), Name="OptionText",
          }, opt)
          ThemeBind(optLbl,"TextColor3","Text")
          local optBtn=Custom:Create("TextButton",{
            Font=Enum.Font.SourceSans, Text="", BackgroundTransparency=0.999,
            BorderSizePixel=0, Size=UDim2.new(1,0,1,0),
          }, opt)
          optBtn.Activated:Connect(function()
            CircleClick(optBtn,Player:GetMouse().X,Player:GetMouse().Y)
            local selected=opt.BackgroundTransparency<0.9
            if Multi then
              if selected then
                for i,v in ipairs(DFuncs.Value) do if v==optName then table.remove(DFuncs.Value,i); break end end
              else
                if not table.find(DFuncs.Value,optName) then table.insert(DFuncs.Value,optName) end
              end
            else DFuncs.Value={optName} end
            DFuncs:Set(DFuncs.Value)
          end)
          local function UpdateCanvas()
            local h=28
            for _,c in ipairs(ScrollSelect:GetChildren()) do
              if c.Name=="DropOption" or c.ClassName=="TextBox" then h=h+c.Size.Y.Offset+3 end
            end
            ScrollSelect.CanvasSize=UDim2.new(0,0,0,h)
          end
          UpdateCanvas(); DropCount+=1
        end
        function DFuncs:RemoveOption(optName)
          for _,child in pairs(ScrollSelect:GetChildren()) do
            if child.Name=="DropOption" then
              local lb=child:FindFirstChild("OptionText")
              if lb and lb.Text==optName then child:Destroy(); break end
            end
          end
          for i,v in ipairs(DFuncs.Value) do if v==optName then table.remove(DFuncs.Value,i); break end end
          DFuncs:Set(DFuncs.Value)
        end
        function DFuncs:Refresh(list,selecting)
          list=list or {}; selecting=selecting or {}
          DFuncs:Clear()
          for _,n in ipairs(list) do DFuncs:AddOption(n) end
          DFuncs.Options=list; DFuncs:Set(selecting)
        end
        DropBtn.Activated:Connect(function()
          if not MoreBlur.Visible then
            MoreBlur.Visible=true
            local pl=DropdownFolder:FindFirstChildOfClass("UIPageLayout")
            if pl then pl:JumpToIndex(SelectBox.LayoutOrder) end
            local tw=TweenInfo.new(0.15)
            TweenService:Create(MoreBlur,tw,{BackgroundTransparency=0.6}):Play()
            TweenService:Create(DropdownSelect,tw,{Position=UDim2.new(1,-11,0.5,0)}):Play()
            TweenService:Create(SelStroke,tw,{Color=Theme.Accent,Transparency=0}):Play()
          end
        end)
        ConnectButton.Activated:Connect(function()
          TweenService:Create(SelStroke,TweenInfo.new(0.12),{Color=Theme.Stroke,Transparency=0.4}):Play()
        end)
        DFuncs:Refresh(DFuncs.Options,DFuncs.Value)
        ItemCount+=1; CountDropdown+=1; return DFuncs
      end

      ----------- ThemePicker -----------
      function Item:AddThemePicker(Config)
        Config=Config or {}
        local saved=Config.SaveKey and SaveSystem:Get(Config.SaveKey)
        local initT=saved or "Crimson"
        local df=Item:AddDropdown({
          Config[1] or Config.Title or "Theme",
          Config[2] or Config.Content or "Choose a color theme",
          false, MIXUSFIELD_Library:GetThemeNames(), {initT},
          function(val) if val and val[1] then MIXUSFIELD_Library:SetTheme(val[1]) end end,
          Config.SaveKey,
        })
        MIXUSFIELD_Library:SetTheme(initT); return df
      end

      function Item:Badge(n)
        if n and n>0 then SecBadgeLbl.Text=tostring(n); SecBadge.Visible=true
        else SecBadge.Visible=false end
      end

      CountSection+=1
      return Item
    end -- AddSection

    CountTab+=1
    return Sections
  end -- CreateTab

  return Tabs
end -- CreateWindow
