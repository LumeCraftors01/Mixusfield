local Utility = {}

function Utility.Tween(instance, info, properties)
    if not instance or not instance.Parent then return end
    local ok, tween = pcall(function()
        return TweenService:Create(instance, info, properties)
    end)
    if ok and tween then
        tween:Play()
        return tween
    end
end

function Utility.TI(t, style, dir, rep, rev, delay)
    return TweenInfo.new(
        t or 0.25,
        style or Enum.EasingStyle.Quart,
        dir or Enum.EasingDirection.Out,
        rep or 0,
        rev or false,
        delay or 0
    )
end

function Utility.Hex(hex)
    hex = hex:gsub("#", "")
    local r = tonumber(hex:sub(1,2), 16) / 255
    local g = tonumber(hex:sub(3,4), 16) / 255
    local b = tonumber(hex:sub(5,6), 16) / 255
    return Color3.new(r, g, b)
end

function Utility.LerpColor(c1, c2, t)
    return Color3.new(
        c1.R + (c2.R - c1.R) * t,
        c1.G + (c2.G - c1.G) * t,
        c1.B + (c2.B - c1.B) * t
    )
end

function Utility.ViewportSize()
    local vp = Camera.ViewportSize
    return vp.X, vp.Y
end

function Utility.IsMobile()
    return UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled
end

function Utility.Scale(value)
    local vx, _ = Utility.ViewportSize()
    return math.clamp(value * (vx / 1920), value * 0.45, value * 1.6)
end

function Utility.ClampPosition(frame, padding)
    padding = padding or 0
    local vx, vy = Utility.ViewportSize()
    local sx = frame.AbsoluteSize.X
    local sy = frame.AbsoluteSize.Y
    local px = math.clamp(frame.AbsolutePosition.X, padding, vx - sx - padding)
    local py = math.clamp(frame.AbsolutePosition.Y, padding, vy - sy - padding)
    frame.Position = UDim2.fromOffset(px, py)
end

function Utility.DeepCopy(tbl)
    local copy = {}
    for k, v in pairs(tbl) do
        if type(v) == "table" then
            copy[k] = Utility.DeepCopy(v)
        else
            copy[k] = v
        end
    end
    return copy
end

function Utility.SafeCall(fn, ...)
    if type(fn) ~= "function" then return end
    local ok, err = pcall(fn, ...)
    if not ok then
        warn("[Mixusfield] Callback error: " .. tostring(err))
    end
end

function Utility.UID()
    return HttpService:GenerateGUID(false):sub(1, 8)
end

function Utility.Spring(instance, props, stiffness, damping, speed)
    stiffness = stiffness or 200
    damping   = damping   or 18
    speed     = speed     or 60
    local dt  = 1 / speed

    local function lerp(a, b, t) return a + (b - a) * t end

    return task.spawn(function()
        local velocities = {}
        local currents   = {}

        for k, target in pairs(props) do
            if typeof(target) == "number" then
                currents[k]   = instance[k] or 0
                velocities[k] = 0
            elseif typeof(target) == "Color3" then
                local c = instance[k] or Color3.new()
                currents[k]   = { c.R, c.G, c.B }
                velocities[k] = { 0, 0, 0 }
            end
        end

        local MAX_STEPS = speed * 3
        local step = 0
        while step < MAX_STEPS do
            step = step + 1
            local settled = true
            for k, target in pairs(props) do
                if typeof(target) == "number" then
                    local x = currents[k]
                    local v = velocities[k]
                    local force = stiffness * (target - x) - damping * v
                    v = v + force * dt
                    x = x + v * dt
                    velocities[k] = v
                    currents[k]   = x
                    pcall(function() instance[k] = x end)
                    if math.abs(target - x) > 0.001 or math.abs(v) > 0.001 then settled = false end
                elseif typeof(target) == "Color3" then
                    for i = 1, 3 do
                        local x = currents[k][i]
                        local v = velocities[k][i]
                        local t = i == 1 and target.R or i == 2 and target.G or target.B
                        local force = stiffness * (t - x) - damping * v
                        v = v + force * dt
                        x = x + v * dt
                        velocities[k][i] = v
                        currents[k][i]   = x
                        if math.abs(t - x) > 0.001 or math.abs(v) > 0.001 then settled = false end
                    end
                    pcall(function()
                        instance[k] = Color3.new(
                            math.clamp(currents[k][1],0,1),
                            math.clamp(currents[k][2],0,1),
                            math.clamp(currents[k][3],0,1)
                        )
                    end)
                end
            end
            if settled then break end
            task.wait(dt)
        end
        for k, target in pairs(props) do
            pcall(function() instance[k] = target end)
        end
    end)
end

function Utility.Stagger(frames, delay, animFn)
    delay  = delay  or 0.04
    animFn = animFn or function(f, i)
        f.BackgroundTransparency = 1
        Utility.Tween(f, Utility.TI(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
            BackgroundTransparency = 0,
        })
    end
    task.spawn(function()
        for i, f in ipairs(frames) do
            if f and f.Parent then
                task.spawn(animFn, f, i)
            end
            task.wait(delay)
        end
    end)
end

function Utility.HSVtoRGB(h, s, v)
    if s == 0 then return Color3.new(v, v, v) end
    h = h * 6
    local i = math.floor(h)
    local f = h - i
    local p = v * (1 - s)
    local q = v * (1 - s * f)
    local t = v * (1 - s * (1 - f))
    local r, g, b
    local idx = i % 6
    if     idx == 0 then r,g,b = v,t,p
    elseif idx == 1 then r,g,b = q,v,p
    elseif idx == 2 then r,g,b = p,v,t
    elseif idx == 3 then r,g,b = p,q,v
    elseif idx == 4 then r,g,b = t,p,v
    elseif idx == 5 then r,g,b = v,p,q
    end
    return Color3.new(r, g, b)
end

function Utility.ColorToHex(c)
    return string.format("#%02X%02X%02X",
        math.floor(c.R * 255 + 0.5),
        math.floor(c.G * 255 + 0.5),
        math.floor(c.B * 255 + 0.5)
    )
end

local IconSystem = {}

function IconSystem.Create(parent, iconDef, size, color, zIndex)
    if not iconDef then return nil end
    local sz  = size or UDim2.fromOffset(18, 18)
    local col = color or Color3.new(1,1,1)
    local z   = zIndex or 5

    if iconDef.Type == "Asset" then
        local img = Instance.new("ImageLabel")
        img.Name                 = "Icon"
        img.BackgroundTransparency = 1
        img.Size                 = sz
        img.ImageColor3          = col
        img.ZIndex               = z
        img.Image                = iconDef.Source
        img.Parent               = parent
        return img
    else
        local src = iconDef.Source or ""
        local glyph = Icons[src:lower()] or src
        local lbl = Instance.new("TextLabel")
        lbl.Name                 = "Icon"
        lbl.BackgroundTransparency = 1
        lbl.Size                 = sz
        lbl.TextColor3           = col
        lbl.Text                 = glyph or "•"
        lbl.Font                 = Enum.Font.GothamBold
        lbl.TextSize             = math.max(sz.Y.Offset - 4, 10)
        lbl.TextScaled           = false
        lbl.TextXAlignment       = Enum.TextXAlignment.Center
        lbl.TextYAlignment       = Enum.TextYAlignment.Center
        lbl.ZIndex               = z
        lbl.Parent               = parent
        return lbl
    end
end

local Builder = {}

function Builder.New(class, props, parent)
    local inst = Instance.new(class)
    for k, v in pairs(props or {}) do
        inst[k] = v
    end
    if parent then inst.Parent = parent end
    return inst
end

function Builder.Frame(props, parent)
    props.BackgroundColor3 = props.BackgroundColor3 or Color3.new(0,0,0)
    return Builder.New("Frame", props, parent)
end

function Builder.ScrollFrame(props, parent)
    return Builder.New("ScrollingFrame", props, parent)
end

function Builder.Label(props, parent)
    props.BackgroundTransparency = props.BackgroundTransparency or 1
    props.TextXAlignment = props.TextXAlignment or Enum.TextXAlignment.Left
    props.Font           = props.Font or Enum.Font.Gotham
    return Builder.New("TextLabel", props, parent)
end

function Builder.TextButton(props, parent)
    props.Font           = props.Font or Enum.Font.GothamSemibold
    props.AutoButtonColor = false
    return Builder.New("TextButton", props, parent)
end

function Builder.TextBox(props, parent)
    props.Font = props.Font or Enum.Font.Gotham
    props.BackgroundTransparency = props.BackgroundTransparency or 1
    return Builder.New("TextBox", props, parent)
end

function Builder.Corner(radius, parent)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, radius or 8)
    c.Parent = parent
    return c
end

function Builder.Stroke(color, thickness, parent)
    local s = Instance.new("UIStroke")
    s.Color     = color or Color3.new(1,1,1)
    s.Thickness = thickness or 1
    s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    s.Parent    = parent
    return s
end

function Builder.Padding(top, right, bottom, left, parent)
    local p = Instance.new("UIPadding")
    p.PaddingTop    = UDim.new(0, top    or 0)
    p.PaddingRight  = UDim.new(0, right  or 0)
    p.PaddingBottom = UDim.new(0, bottom or 0)
    p.PaddingLeft   = UDim.new(0, left   or 0)
    p.Parent = parent
    return p
end

function Builder.ListLayout(props, parent)
    local l = Instance.new("UIListLayout")
    for k,v in pairs(props or {}) do l[k] = v end
    l.Parent = parent
    return l
end

function Builder.Gradient(colorSeq, rotation, parent)

            end)
            TooltipSystem._tip = nil
        end
    end

    showConn = obj.MouseEnter:Connect(function()
        task.delay(0.4, function()
            if not TooltipSystem._tip then showTip() end
        end)
    end)
    hideConn = obj.MouseLeave:Connect(hideTip)
end

    Utility.Tween(progressFill, Utility.TI(duration, Enum.EasingStyle.Linear), { Size = UDim2.new(0, 0, 1, 0) })

    -- Auto dismiss
    task.delay(duration, dismissNotif)

local ThemeManager = {}
ThemeManager._listeners = {}
ThemeManager._current   = nil

function ThemeManager.Set(themeName, accent)
    local base = Utility.DeepCopy(Themes[themeName] or Themes.Dark)
    if accent then
        base.Accent    = accent
        base.AccentDim = Utility.LerpColor(accent, Color3.new(0,0,0), 0.25)
        base.AccentGlow = accent
        base.AccentText = accent
    end
    ThemeManager._current = base
    for _, fn in ipairs(ThemeManager._listeners) do
        Utility.SafeCall(fn, base)
    end
    return base
end

function ThemeManager.Get()
    return ThemeManager._current or ThemeManager.Set("Dark")
end

function ThemeManager.OnChanged(fn)
    table.insert(ThemeManager._listeners, fn)
end

-- [[ SEARCH SYSTEM ]]
-- ─────────────────────────────────────────────────────────────────────────────
local SearchRegistry = {}
SearchRegistry._entries = {}

function SearchRegistry.Register(tabId, componentFrame, nameText)
    table.insert(SearchRegistry._entries, {
        TabId = tabId,
        Frame = componentFrame,
        Name  = nameText:lower(),
    })
end

function SearchRegistry.Filter(query)
    query = query:lower():gsub("%s+", "")
    local empty = (query == "")
    for _, entry in ipairs(SearchRegistry._entries) do
        if entry.Frame and entry.Frame.Parent then
            if empty then
                entry.Frame.Visible = true
            else
                local match = entry.Name:find(query, 1, true)
                entry.Frame.Visible = match ~= nil
            end
        end
    end
end

function SearchRegistry.Clear()
    SearchRegistry._entries = {}
end

local FloatingButton = {}

function FloatingButton.Create(screenGui, iconDef, onToggle, theme)
    local BTN_SIZE    = 44
    local SNAP_PADDING = 12

    local btn = Builder.Frame({
        Name              = "MixusfieldToggleButton",
        BackgroundColor3  = theme.Accent,
        Size              = UDim2.fromOffset(BTN_SIZE, BTN_SIZE),
        Position          = UDim2.new(0, SNAP_PADDING, 0.5, -BTN_SIZE/2),
        ZIndex            = 200,
        Active            = true,
    }, screenGui)
    Builder.Corner(12, btn)
    Builder.Shadow(btn, 20, theme.AccentGlow, 0.4)

    local iconLabel = IconSystem.Create(btn, iconDef or { Type="Lucide", Source="menu" },
        UDim2.fromOffset(22, 22), Color3.new(1,1,1), 201)
    if iconLabel then
        iconLabel.AnchorPoint = Vector2.new(0.5, 0.5)
        iconLabel.Position    = UDim2.fromOffset(BTN_SIZE/2, BTN_SIZE/2)
    end

    local glowFrame = Builder.Frame({
        Name             = "Glow",
        BackgroundColor3 = theme.AccentGlow,
        BackgroundTransparency = 0.7,
        Size             = UDim2.fromOffset(BTN_SIZE + 10, BTN_SIZE + 10),
        Position         = UDim2.fromOffset(-5, -5),
        ZIndex           = 199,
    }, btn)
    Builder.Corner(16, glowFrame)

    task.spawn(function()
        while btn and btn.Parent do
            Utility.Tween(glowFrame, Utility.TI(0.9, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                { BackgroundTransparency = 0.9 })
            task.wait(0.9)
            Utility.Tween(glowFrame, Utility.TI(0.9, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut),
                { BackgroundTransparency = 0.7 })
            task.wait(0.9)
        end
    end)

    local dragging  = false
    local dragStart = Vector2.new(0, 0)
    local startPos  = Vector2.new(0, 0)
    local moved     = false
    local DRAG_THRESHOLD = 6

    local function snapToEdge()
        local vx, vy = Utility.ViewportSize()
        local cx = btn.AbsolutePosition.X + BTN_SIZE / 2
        if cx < vx / 2 then
            Utility.Tween(btn, Utility.TI(0.35, Enum.EasingStyle.Back), {
                Position = UDim2.fromOffset(SNAP_PADDING, btn.AbsolutePosition.Y)
            })
        else
            Utility.Tween(btn, Utility.TI(0.35, Enum.EasingStyle.Back), {
                Position = UDim2.fromOffset(vx - BTN_SIZE - SNAP_PADDING, btn.AbsolutePosition.Y)
            })
        end
    end

    local conns = ConnectionManager.new()

    local function onInputBegan(input)
        local itype = input.UserInputType
        if itype == Enum.UserInputType.MouseButton1 or itype == Enum.UserInputType.Touch then
            dragging  = true
            moved     = false
            dragStart = Vector2.new(input.Position.X, input.Position.Y)
            startPos  = Vector2.new(btn.AbsolutePosition.X, btn.AbsolutePosition.Y)
            Utility.Tween(btn, Utility.TI(0.1), { Size = UDim2.fromOffset(BTN_SIZE - 4, BTN_SIZE - 4) })
        end
    end

    local function onInputChanged(input)
        if not dragging then return end
        local itype = input.UserInputType
        if itype == Enum.UserInputType.MouseMovement or itype == Enum.UserInputType.Touch then
            local delta = Vector2.new(input.Position.X - dragStart.X, input.Position.Y - dragStart.Y)
            if delta.Magnitude > DRAG_THRESHOLD then moved = true end
            if moved then
                local vx, vy = Utility.ViewportSize()
                local nx = math.clamp(startPos.X + delta.X, SNAP_PADDING, vx - BTN_SIZE - SNAP_PADDING)
                local ny = math.clamp(startPos.Y + delta.Y, SNAP_PADDING, vy - BTN_SIZE - SNAP_PADDING)
                btn.Position = UDim2.fromOffset(nx, ny)
            end
        end
    end

    local function onInputEnded(input)
        local itype = input.UserInputType
        if itype == Enum.UserInputType.MouseButton1 or itype == Enum.UserInputType.Touch then
            Utility.Tween(btn, Utility.TI(0.15), { Size = UDim2.fromOffset(BTN_SIZE, BTN_SIZE) })
            if dragging then
                dragging = false
                if moved then
                    snapToEdge()
                else
                    Utility.SafeCall(onToggle)
                end
            end
        end
    end

    conns:Add(btn.InputBegan:Connect(onInputBegan))
    conns:Add(UserInputService.InputChanged:Connect(onInputChanged))
    conns:Add(UserInputService.InputEnded:Connect(onInputEnded))

    conns:Add(btn.MouseEnter:Connect(function()
        Utility.Tween(btn, Utility.TI(0.15), { BackgroundColor3 = ThemeManager.Get().AccentDim })
    end))
    conns:Add(btn.MouseLeave:Connect(function()
        Utility.Tween(btn, Utility.TI(0.15), { BackgroundColor3 = ThemeManager.Get().Accent })
    end))

    return btn, conns
end

-- ── Slider ────────────────────────────────────────────────────────────────────
function Components.Slider(parent, config, theme, tabId)
    --[[
    config = {
        Name      = "Speed",
        Min       = 0, Max = 100, Default = 50,
        Increment = 1,
        Suffix    = "",
        Callback  = function(value) end,
    }
    ]]
    local COMP_HEIGHT = 58
    local min = config.Min or 0
    local max = config.Max or 100
    local inc = config.Increment or 1
    local val = math.clamp(config.Default or min, min, max)

    local function snap(v)
        return math.clamp(math.floor((v - min) / inc + 0.5) * inc + min, min, max)
    end

    local frame = Builder.Frame({
        Name             = "Slider_" .. (config.Name or Utility.UID()),
        BackgroundColor3 = theme.Surface,
        Size             = UDim2.new(1, 0, 0, COMP_HEIGHT),
        ZIndex           = 5,
    }, parent)
    Builder.Padding(0, 16, 0, 16, frame)

    Builder.Frame({
        BackgroundColor3 = theme.Border,
        Size   = UDim2.new(1, 0, 0, 1),
        Position = UDim2.new(0, 0, 1, -1),
        ZIndex = 5,
    }, frame)

    Builder.Label({
        Text     = config.Name or "Slider",
        TextColor3 = theme.Text,
        Font     = Enum.Font.GothamSemibold,
        TextSize = 13,
        Size     = UDim2.new(1, -80, 0, 20),
        Position = UDim2.fromOffset(0, 8),
        ZIndex   = 6,
    }, frame)

    -- Value label (right)
    local valueLabel = Builder.Label({
        Text     = tostring(val) .. (config.Suffix or ""),
        TextColor3 = theme.AccentText,
        Font     = Enum.Font.GothamBold,
        TextSize = 12,
        Size     = UDim2.fromOffset(65, 20),
        Position = UDim2.new(1, -65, 0, 8),
        TextXAlignment = Enum.TextXAlignment.Right,
        ZIndex   = 6,
    }, frame)

    -- Track
    local trackBg = Builder.Frame({
        Name             = "SliderTrack",
        BackgroundColor3 = theme.SliderTrack,
        Size             = UDim2.new(1, 0, 0, 5),
        Position         = UDim2.new(0, 0, 1, -17),
        ZIndex           = 6,
    }, frame)
    Builder.Corner(3, trackBg)

    local trackFill = Builder.Frame({
        Name             = "SliderFill",
        BackgroundColor3 = theme.Accent,
        Size             = UDim2.new((val - min) / (max - min), 0, 1, 0),
        ZIndex           = 7,
    }, trackBg)
    Builder.Corner(3, trackFill)

    -- Thumb
    local thumb = Builder.Frame({
        Name             = "SliderThumb",
        BackgroundColor3 = Color3.new(1,1,1),
        Size             = UDim2.fromOffset(14, 14),
        AnchorPoint      = Vector2.new(0.5, 0.5),
        Position         = UDim2.new((val - min) / (max - min), 0, 0.5, 0),
        ZIndex           = 8,
    }, trackBg)
    Builder.Corner(7, thumb)
    Builder.Stroke(theme.Accent, 2, thumb)

    local draggingSlider = false
    local sliderConns = ConnectionManager.new()

    local function setSliderValue(rawX)
        local trackAbsX = trackBg.AbsolutePosition.X
        local trackAbsW = trackBg.AbsoluteSize.X
        if trackAbsW == 0 then return end
        local t = math.clamp((rawX - trackAbsX) / trackAbsW, 0, 1)
        local newVal = snap(min + t * (max - min))
        if newVal ~= val then
            val = newVal
            local pct = (val - min) / (max - min)
            trackFill.Size = UDim2.new(pct, 0, 1, 0)
            thumb.Position = UDim2.new(pct, 0, 0.5, 0)
            valueLabel.Text = tostring(val) .. (config.Suffix or "")
            Utility.SafeCall(config.Callback, val)
        end
    end

    local sliderBtn = Builder.TextButton({
        Text="", BackgroundTransparency=1,
        Size=UDim2.new(1, 0, 0, 30),
        Position=UDim2.new(0, 0, 1, -30),
        ZIndex=9,
    }, frame)

    sliderConns:Add(sliderBtn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            draggingSlider = true
            setSliderValue(input.Position.X)
            Utility.Tween(thumb, Utility.TI(0.1), { Size = UDim2.fromOffset(18, 18) })
        end
    end))
    sliderConns:Add(UserInputService.InputChanged:Connect(function(input)
        if not draggingSlider then return end
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            setSliderValue(input.Position.X)
        end
    end))
    sliderConns:Add(UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            if draggingSlider then
                draggingSlider = false
                Utility.Tween(thumb, Utility.TI(0.12), { Size = UDim2.fromOffset(14, 14) })
            end
        end
    end))

    frame.Destroying:Connect(function() sliderConns:Clean() end)
    SearchRegistry.Register(tabId, frame, config.Name or "Slider")

    local api = {}
    function api:Set(newVal)
        val = snap(newVal)
        local pct = (val - min) / (max - min)
        trackFill.Size = UDim2.new(pct, 0, 1, 0)
        thumb.Position = UDim2.new(pct, 0, 0.5, 0)
        valueLabel.Text = tostring(val) .. (config.Suffix or "")
    end
    function api:Get() return val end

    return frame, api
end

        Size             = UDim2.new(1, 0, 1, -42),
        Position         = UDim2.fromOffset(0, 42),
        ScrollBarThickness = 3,
        ScrollBarImageColor3 = theme.ScrollBar,
        CanvasSize       = UDim2.fromOffset(0, 0),
        AutomaticCanvasSize = Enum.AutomaticSize.Y,
        ZIndex           = 51,
    }, dropPanel)
    Builder.Padding(4, 6, 4, 6, itemsScroll)
    Builder.ListLayout({ Padding = UDim.new(0, 2), SortOrder = Enum.SortOrder.LayoutOrder }, itemsScroll)

    local itemButtons = {}

    local function populateItems(filter)
        -- Clear
        for _, ch in ipairs(itemsScroll:GetChildren()) do
            if ch:IsA("TextButton") or ch:IsA("Frame") then
                ch:Destroy()
            end
        end
        itemButtons = {}

        local opts = config.Options or {}
        for i, opt in ipairs(opts) do
            if filter == "" or opt:lower():find(filter:lower(), 1, true) then
                local row = Builder.Frame({
                    BackgroundColor3 = (opt == selected) and theme.SidebarActive or theme.SurfaceAlt,
                    Size             = UDim2.new(1, 0, 0, ITEM_HEIGHT),
                    ZIndex           = 52,
                    LayoutOrder      = i,
                }, itemsScroll)
                Builder.Corner(6, row)
                Builder.Padding(0, 8, 0, 8, row)

                -- Check mark for selected
                if opt == selected then
                    Builder.Label({
                        Text       = "✓",
                        TextColor3 = theme.AccentText,
                        Font       = Enum.Font.GothamBold,
                        TextSize   = 11,
                        Size       = UDim2.fromOffset(16, ITEM_HEIGHT),
                        ZIndex     = 53,
                        TextXAlignment = Enum.TextXAlignment.Center,
                    }, row)
                end

                Builder.Label({
                    Text       = opt,
                    TextColor3 = (opt == selected) and theme.Text or theme.TextDim,
                    Font       = (opt == selected) and Enum.Font.GothamSemibold or Enum.Font.Gotham,
                    TextSize   = 13,
                    Size       = UDim2.new(1, -20, 1, 0),
                    Position   = UDim2.fromOffset(20, 0),
                    ZIndex     = 53,
                }, row)

                local rowBtn = Builder.TextButton({
                    Text="", BackgroundTransparency=1, Size=UDim2.fromScale(1,1), ZIndex=54,
                }, row)

                local capturedOpt = opt
                rowBtn.MouseButton1Click:Connect(function()
                    selected = capturedOpt
                    selectedLabel.Text = selected
                    selectedLabel.TextColor3 = ThemeManager.Get().TextDim
                    Utility.SafeCall(config.Callback, selected)
                    -- Close
                    isOpen = false
                    Utility.Tween(dropPanel, Utility.TI(0.2, Enum.EasingStyle.Quart), { Size = UDim2.fromOffset(200, 0) })
                    task.delay(0.22, function()
                        if dropPanel and dropPanel.Parent then
                            dropPanel.Visible = false
                        end
                    end)
                    Utility.Tween(chevron, Utility.TI(0.2), { Rotation = 0 })
                    -- Repopulate to update checkmarks
                    task.defer(function() populateItems("") end)
                end)
                rowBtn.MouseEnter:Connect(function()
                    if capturedOpt ~= selected then
                        Utility.Tween(row, Utility.TI(0.1), { BackgroundColor3 = ThemeManager.Get().SidebarHover })
                    end
                end)
                rowBtn.MouseLeave:Connect(function()
                    if capturedOpt ~= selected then
                        Utility.Tween(row, Utility.TI(0.1), { BackgroundColor3 = ThemeManager.Get().SurfaceAlt })
                    end
                end)

                table.insert(itemButtons, row)
            end
        end
    end

    populateItems("")

    searchBox:GetPropertyChangedSignal("Text"):Connect(function()
        populateItems(searchBox.Text)
    end)

    -- Calculate panel height
    local function getPanelHeight()
        local opts = config.Options or {}
        local count = math.min(#opts, 6)
        return 42 + count * (ITEM_HEIGHT + 2) + 10
    end

    -- Toggle open/close
    local mainBtn = Builder.TextButton({
        Text="", BackgroundTransparency=1, Size=UDim2.fromScale(1,1), ZIndex=8,
    }, frame)

    mainBtn.MouseButton1Click:Connect(function()
        isOpen = not isOpen
        if isOpen then
            dropPanel.Visible = true
            dropPanel.Size = UDim2.fromOffset(200, 0)
            searchBox.Text = ""
            populateItems("")
            Utility.Tween(dropPanel, Utility.TI(0.22, Enum.EasingStyle.Quart), { Size = UDim2.fromOffset(200, getPanelHeight()) })
            Utility.Tween(chevron, Utility.TI(0.2), { Rotation = 180 })
        else
            Utility.Tween(dropPanel, Utility.TI(0.18, Enum.EasingStyle.Quart), { Size = UDim2.fromOffset(200, 0) })
            task.delay(0.2, function()
                if dropPanel and dropPanel.Parent then
                    dropPanel.Visible = false
                end
            end)
            Utility.Tween(chevron, Utility.TI(0.2), { Rotation = 0 })
        end
    end)
    mainBtn.MouseEnter:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().SurfaceAlt }) end)
    mainBtn.MouseLeave:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().Surface }) end)

    SearchRegistry.Register(tabId, frame, config.Name or "Dropdown")

    local api = {}
    function api:Set(val)
        selected = val
        selectedLabel.Text = selected
        selectedLabel.TextColor3 = ThemeManager.Get().TextDim
        task.defer(function() populateItems("") end)
    end
    function api:Get() return selected end
    function api:Refresh(newOpts)
        config.Options = newOpts
        task.defer(function() populateItems("") end)
    end

    return frame, api
end

-- ── ColorPicker ───────────────────────────────────────────────────────────────
function Components.ColorPicker(parent, config, theme, tabId)
    local COMP_HEIGHT = 46
    local currentColor = config.Default or Color3.fromRGB(124, 107, 240)
    local isOpen = false

    local frame = Builder.Frame({
        Name             = "ColorPicker_" .. Utility.UID(),
        BackgroundColor3 = theme.Surface,
        Size             = UDim2.new(1, 0, 0, COMP_HEIGHT),
        ZIndex           = 5,
        ClipsDescendants = false,
    }, parent)
    Builder.Padding(0, 16, 0, 16, frame)

    Builder.Frame({
        BackgroundColor3 = theme.Border,
        Size   = UDim2.new(1, 0, 0, 1),
        Position = UDim2.new(0, 0, 1, -1),
        ZIndex = 5,
    }, frame)

    Builder.Label({
        Text     = config.Name or "Color",
        TextColor3 = theme.Text,
        Font     = Enum.Font.GothamSemibold,
        TextSize = 13,
        Size     = UDim2.new(1, -70, 1, 0),
        ZIndex   = 6,
    }, frame)

    -- Color swatch preview
    local swatch = Builder.Frame({
        BackgroundColor3 = currentColor,
        Size             = UDim2.fromOffset(44, 24),
        Position         = UDim2.new(1, -46, 0.5, -12),
        ZIndex           = 6,
    }, frame)
    Builder.Corner(6, swatch)
    Builder.Stroke(theme.Border, 1, swatch)

    -- Color picker panel
    local pickerPanel = Builder.Frame({
        Name             = "ColorPickerPanel",
        BackgroundColor3 = theme.SurfaceAlt,
        Size             = UDim2.fromOffset(220, 0),
        Position         = UDim2.new(1, -222, 1, 4),
        ZIndex           = 45,
        Visible          = false,
        ClipsDescendants = true,
    }, frame)
    Builder.Corner(10, pickerPanel)
    Builder.Stroke(theme.Border, 1, pickerPanel)
    Builder.Shadow(pickerPanel, 16, theme.Shadow, 0.55)
    Builder.Padding(12, 12, 12, 12, pickerPanel)

    local pickerLayout = Builder.ListLayout({ Padding = UDim.new(0, 8), SortOrder = Enum.SortOrder.LayoutOrder }, pickerPanel)

    -- Hue, Saturation, Value sliders
    local h, s, v = 0, 1, 1
    pcall(function()
        h, s, v = currentColor:ToHSV()
    end)

    local sliderData = {
        { label = "H", val = h, key = "h" },
        { label = "S", val = s, key = "s" },
        { label = "V", val = v, key = "v" },
    }

    local hsv = { h = h, s = s, v = v }

    local function updateColor()
        currentColor = Utility.HSVtoRGB(hsv.h, hsv.s, hsv.v)
        swatch.BackgroundColor3 = currentColor
        Utility.SafeCall(config.Callback, currentColor)
    end

    local pickerConns = ConnectionManager.new()

    for i, data in ipairs(sliderData) do
        local row = Builder.Frame({
            BackgroundTransparency = 1,
            Size   = UDim2.new(1, 0, 0, 28),
            LayoutOrder = i,
            ZIndex = 46,
        }, pickerPanel)
        Builder.Padding(0, 0, 0, 0, row)

        Builder.Label({
            Text     = data.label,
            TextColor3 = theme.TextDim,
            Font     = Enum.Font.GothamSemibold,
            TextSize = 11,
            Size     = UDim2.fromOffset(14, 28),
            ZIndex   = 47,
        }, row)

        local trackBg = Builder.Frame({
            BackgroundColor3 = theme.SliderTrack,
            Size   = UDim2.new(1, -50, 0, 5),
            Position = UDim2.new(0, 18, 0.5, -2),
            ZIndex = 47,
        }, row)
        Builder.Corner(3, trackBg)

        local trackFill = Builder.Frame({
            BackgroundColor3 = theme.Accent,
            Size   = UDim2.new(data.val, 0, 1, 0),
            ZIndex = 48,
        }, trackBg)
        Builder.Corner(3, trackFill)

        local thumb2 = Builder.Frame({
            BackgroundColor3 = Color3.new(1,1,1),
            Size   = UDim2.fromOffset(12, 12),
            AnchorPoint = Vector2.new(0.5, 0.5),
            Position = UDim2.new(data.val, 0, 0.5, 0),
            ZIndex = 49,
        }, trackBg)
        Builder.Corner(6, thumb2)

        local valLabel = Builder.Label({
            Text     = string.format("%.2f", data.val),
            TextColor3 = theme.TextMuted,
            Font     = Enum.Font.Gotham,
            TextSize = 10,
            Size     = UDim2.fromOffset(30, 28),
            Position = UDim2.new(1, -32, 0, 0),
            ZIndex   = 47,
            TextXAlignment = Enum.TextXAlignment.Right,
        }, row)

        local dragging2 = false
        local capturedKey = data.key

        local sBtn = Builder.TextButton({
            Text="", BackgroundTransparency=1,
            Size=UDim2.new(1,-50,0,20),
            Position=UDim2.new(0,18,0.5,-10),
            ZIndex=50,
        }, row)

        pickerConns:Add(sBtn.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging2 = true
            end
        end))
        pickerConns:Add(UserInputService.InputChanged:Connect(function(input)
            if not dragging2 then return end
            if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
                local trackAbsX = trackBg.AbsolutePosition.X
                local trackAbsW = trackBg.AbsoluteSize.X
                if trackAbsW == 0 then return end
                local t = math.clamp((input.Position.X - trackAbsX) / trackAbsW, 0, 1)
                hsv[capturedKey] = t
                trackFill.Size = UDim2.new(t, 0, 1, 0)
                thumb2.Position = UDim2.new(t, 0, 0.5, 0)
                valLabel.Text = string.format("%.2f", t)
                updateColor()
            end
        end))
        pickerConns:Add(UserInputService.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging2 = false
            end
        end))
    end

    -- Hex display
    local hexLabel = Builder.Label({
        Text     = Utility.ColorToHex(currentColor),
        TextColor3 = theme.TextMuted,
        Font     = Enum.Font.GothamSemibold,
        TextSize = 11,
        Size     = UDim2.new(1, 0, 0, 16),
        LayoutOrder = 4,
        ZIndex   = 46,
        TextXAlignment = Enum.TextXAlignment.Center,
    }, pickerPanel)

    -- We override updateColor to also update hex label
    local origUpdate = updateColor
    updateColor = function()
        origUpdate()
        hexLabel.Text = Utility.ColorToHex(currentColor)
    end

    frame.Destroying:Connect(function() pickerConns:Clean() end)

    local swatchBtn = Builder.TextButton({
        Text="", BackgroundTransparency=1, Size=UDim2.fromScale(1,1), ZIndex=8,
    }, frame)

    swatchBtn.MouseButton1Click:Connect(function()
        isOpen = not isOpen
        if isOpen then
            pickerPanel.Visible = true
            pickerPanel.Size = UDim2.fromOffset(220, 0)
            Utility.Tween(pickerPanel, Utility.TI(0.22, Enum.EasingStyle.Quart), { Size = UDim2.fromOffset(220, 130) })
        else
            Utility.Tween(pickerPanel, Utility.TI(0.18, Enum.EasingStyle.Quart), { Size = UDim2.fromOffset(220, 0) })
            task.delay(0.2, function()
                if pickerPanel and pickerPanel.Parent then pickerPanel.Visible = false end
            end)
        end
    end)
    swatchBtn.MouseEnter:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().SurfaceAlt }) end)
    swatchBtn.MouseLeave:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().Surface }) end)

    SearchRegistry.Register(tabId, frame, config.Name or "ColorPicker")

    local api = {}
    function api:Set(color)
        currentColor = color
        swatch.BackgroundColor3 = color
        pcall(function() h, s, v = color:ToHSV() hsv.h=h hsv.s=s hsv.v=v end)
    end
    function api:Get() return currentColor end

    return frame, api
end

    local headerRow = Builder.Frame({
        BackgroundTransparency = 1,
        Size   = UDim2.new(1, 0, 0, HEADER_H),
        ZIndex = 6,
    }, frame)

    if config.Icon then
        local ih = Builder.Frame({ BackgroundTransparency=1, Size=UDim2.fromOffset(20, HEADER_H), ZIndex=7 }, headerRow)
        Builder.New("UIListLayout", { HorizontalAlignment=Enum.HorizontalAlignment.Center, VerticalAlignment=Enum.VerticalAlignment.Center }, ih)
        IconSystem.Create(ih, config.Icon, UDim2.fromOffset(16, 16), theme.TextDim, 7)
    end

    Builder.Label({
        Text     = config.Name or "Accordion",
        TextColor3 = theme.Text,
        Font     = Enum.Font.GothamSemibold,
        TextSize = 13,
        Size     = UDim2.new(1, -40, 1, 0),
        Position = UDim2.fromOffset(config.Icon and 24 or 0, 0),
        ZIndex   = 7,
    }, headerRow)

    local arrow = Builder.Label({
        Text       = "▾",
        TextColor3 = theme.TextMuted,
        Font       = Enum.Font.GothamBold,
        TextSize   = 12,
        Size       = UDim2.fromOffset(20, HEADER_H),
        Position   = UDim2.new(1, -20, 0, 0),
        ZIndex     = 7,
        TextXAlignment = Enum.TextXAlignment.Center,
        Rotation   = isOpen and 180 or 0,
    }, headerRow)

    local contentArea = Builder.Frame({
        BackgroundTransparency = 1,
        Size   = UDim2.new(1, 0, 1, -HEADER_H),
        Position = UDim2.fromOffset(0, HEADER_H),
        ZIndex = 6,
        ClipsDescendants = true,
    }, frame)
    Builder.Padding(4, 0, 4, 0, contentArea)
    Builder.ListLayout({ Padding = UDim.new(0, 4), SortOrder = Enum.SortOrder.LayoutOrder }, contentArea)

    local accordionTab = {}
    local layoutOrder  = 0

    local function nextOrder()
        layoutOrder = layoutOrder + 1
        return layoutOrder
    end

    local accordionTheme = theme

    local function makeComp(compName)
        return function(self_tab, cfg)
            cfg = cfg or {}
            cfg._layoutOrder = nextOrder()
            local f, api = Components[compName](contentArea, cfg, accordionTheme, tabId)
            f.LayoutOrder = cfg._layoutOrder
            return api or {}
        end
    end

    accordionTab.CreateButton     = makeComp("Button")
    accordionTab.CreateToggle     = makeComp("Toggle")
    accordionTab.CreateSlider     = makeComp("Slider")
    accordionTab.CreateDropdown   = makeComp("Dropdown")
    accordionTab.CreateTextbox    = makeComp("Textbox")
    accordionTab.CreateKeybind    = makeComp("Keybind")
    accordionTab.CreateLabel      = makeComp("Label")
    accordionTab.CreateParagraph  = makeComp("Paragraph")

    -- Lazy populate
    task.defer(function()
        if config.Content then
            Utility.SafeCall(config.Content, accordionTab)
        end
        local contentH = 0
        for _, child in ipairs(contentArea:GetChildren()) do
            if child:IsA("Frame") or child:IsA("ScrollingFrame") then
                contentH = contentH + (child.AbsoluteSize.Y > 0 and child.AbsoluteSize.Y or 46) + 4
            end
        end
        frame._contentH = contentH + HEADER_H + 12
        if isOpen then
            frame.Size = UDim2.new(1, 0, 0, frame._contentH)
        end
    end)

    local clickBtn = Builder.TextButton({ Text="", BackgroundTransparency=1, Size=UDim2.new(1,0,0,HEADER_H), ZIndex=8 }, frame)
    clickBtn.MouseButton1Click:Connect(function()
        isOpen = not isOpen
        local targetH = isOpen and (frame._contentH or HEADER_H + 60) or HEADER_H
        Utility.Tween(frame, Utility.TI(0.3, Enum.EasingStyle.Quart), { Size = UDim2.new(1, 0, 0, targetH) })
        Utility.Tween(arrow, Utility.TI(0.25, Enum.EasingStyle.Back), { Rotation = isOpen and 180 or 0 })
    end)
    clickBtn.MouseEnter:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().SurfaceAlt }) end)
    clickBtn.MouseLeave:Connect(function() Utility.Tween(frame, Utility.TI(0.12), { BackgroundColor3 = ThemeManager.Get().Surface }) end)

    SearchRegistry.Register(tabId, frame, config.Name or "Accordion")
    return frame, accordionTab
end

-- ── RadioGroup ────────────────────────────────────────────────────────────────
function Components.RadioGroup(parent, config, theme, tabId)
    local ITEM_H  = 40
    local selected = config.Default or (config.Options and config.Options[1]) or ""

    local frame = Builder.Frame({
        Name             = "RadioGroup_" .. Utility.UID(),
        BackgroundColor3 = theme.Surface,
        Size             = UDim2.new(1, 0, 0, 0),
        AutomaticSize    = Enum.AutomaticSize.Y,
        ZIndex           = 5,
    }, parent)
    Builder.Padding(6, 16, 6, 16, frame)

    Builder.Frame({
        BackgroundColor3 = theme.Border,
        Size   = UDim2.new(1, 0, 0, 1),
        Position = UDim2.new(0, 0, 1, -1),
        ZIndex = 5,
    }, frame)

    Builder.Label({
        Text     = config.Name or "Select",
        TextColor3 = theme.TextDim,
        Font     = Enum.Font.GothamBold,
        TextSize = 11,
        Size     = UDim2.new(1, 0, 0, 18),
        LayoutOrder = 0,
        ZIndex   = 6,
    }, frame)

    local inner = Builder.Frame({
        BackgroundTransparency = 1,
        Size   = UDim2.fromScale(1, 0),
        AutomaticSize = Enum.AutomaticSize.Y,
        LayoutOrder = 1,
        ZIndex = 6,
    }, frame)
    Builder.ListLayout({ Padding = UDim.new(0, 2), SortOrder = Enum.SortOrder.LayoutOrder }, inner)
    Builder.ListLayout({ Padding = UDim.new(0, 4), SortOrder = Enum.SortOrder.LayoutOrder }, frame)

    local radioButtons = {}

    local function updateRadios()
        for opt, data in pairs(radioButtons) do
            local active = (opt == selected)
            local outerStroke = data.outer:FindFirstChildWhichIsA("UIStroke")
            if outerStroke then
                Utility.Tween(outerStroke, Utility.TI(0.15), { Color = active and theme.Accent or theme.Border })
            end
            Utility.Tween(data.dot, Utility.TI(0.2, Enum.EasingStyle.Back), {
                Size = active and UDim2.fromOffset(10, 10) or UDim2.fromOffset(0, 0),
                BackgroundColor3 = theme.Accent,
            })
            Utility.Tween(data.label, Utility.TI(0.15), {
                TextColor3 = active and theme.Text or theme.TextDim,
            })
        end
    end

    for i, opt in ipairs(config.Options or {}) do
        local row = Builder.Frame({
            BackgroundColor3 = theme.Surface,
            Size   = UDim2.new(1, 0, 0, ITEM_H),
            LayoutOrder = i,
            ZIndex = 7,
        }, inner)
        Builder.Corner(6, row)

        local outer = Builder.Frame({
            BackgroundColor3 = theme.SurfaceAlt,
            Size   = UDim2.fromOffset(18, 18),
            Position = UDim2.new(0, 0, 0.5, -9),
            ZIndex = 8,
        }, row)
        Builder.Corner(9, outer)
        Builder.Stroke(opt == selected and theme.Accent or theme.Border, 2, outer)

        local dot = Builder.Frame({
            BackgroundColor3 = theme.Accent,
            Size   = opt == selected and UDim2.fromOffset(10, 10) or UDim2.fromOffset(0, 0),
            AnchorPoint = Vector2.new(0.5, 0.5),
            Position = UDim2.fromScale(0.5, 0.5),
            ZIndex = 9,
        }, outer)
        Builder.Corner(5, dot)

        local lbl = Builder.Label({
            Text     = opt,
            TextColor3 = opt == selected and theme.Text or theme.TextDim,
            Font     = Enum.Font.GothamSemibold,
            TextSize = 13,
            Size     = UDim2.new(1, -26, 1, 0),
            Position = UDim2.fromOffset(26, 0),
            ZIndex   = 8,
        }, row)

        radioButtons[opt] = { outer = outer, dot = dot, label = lbl }

        local rb = Builder.TextButton({ Text="", BackgroundTransparency=1, Size=UDim2.fromScale(1,1), ZIndex=10 }, row)
        local capturedOpt = opt
        rb.MouseButton1Click:Connect(function()
            if selected == capturedOpt then return end
            selected = capturedOpt
            updateRadios()
            Utility.SafeCall(config.Callback, selected)
        end)
        rb.MouseEnter:Connect(function() Utility.Tween(row, Utility.TI(0.1), { BackgroundColor3 = ThemeManager.Get().SurfaceAlt }) end)
        rb.MouseLeave:Connect(function() Utility.Tween(row, Utility.TI(0.1), { BackgroundColor3 = ThemeManager.Get().Surface }) end)
    end

    SearchRegistry.Register(tabId, frame, config.Name or "RadioGroup")

    local api = {}
    function api:Set(val)
        selected = val
        updateRadios()
    end
    function api:Get() return selected end

    return frame, api
end

            self:Toggle()

        self._toggleBtn   = toggleBtn
        self._toggleConns = toggleConns
    end

-- ── Window:CreateTab ──────────────────────────────────────────────────────────
function Window:CreateTab(config)
    --[[
    config = {
        Name = "Home",
        Icon = { Type="Lucide", Source="home" },
    }
    ]]
    local theme  = self._theme
    local tabId  = Utility.UID()
    local TAB_H  = 40

    -- Sidebar button
    local sideBtn = Builder.Frame({
        BackgroundColor3 = theme.Sidebar,
        BackgroundTransparency = 1,
        Size   = UDim2.new(1, 0, 0, TAB_H),
        ZIndex = 8,
    }, self._tabScroll)
    Builder.Corner(8, sideBtn)
    Builder.Padding(0, 10, 0, 10, sideBtn)

    local btnRow = Builder.Frame({
        BackgroundTransparency = 1,
        Size   = UDim2.fromScale(1, 1),
        ZIndex = 9,
    }, sideBtn)
    Builder.ListLayout({
        FillDirection = Enum.FillDirection.Horizontal,
        VerticalAlignment = Enum.VerticalAlignment.Center,
        Padding = UDim.new(0, 8),
    }, btnRow)

    local iconEl = nil
    if config.Icon then
        local iconHolder = Builder.Frame({
            BackgroundTransparency = 1,
            Size   = UDim2.fromOffset(18, TAB_H),
            ZIndex = 10,
        }, btnRow)
        Builder.New("UIListLayout", { HorizontalAlignment=Enum.HorizontalAlignment.Center, VerticalAlignment=Enum.VerticalAlignment.Center }, iconHolder)
        iconEl = IconSystem.Create(iconHolder, config.Icon, UDim2.fromOffset(16, 16), theme.TextDim, 10)
    end

    local tabLabel = Builder.Label({
        Text     = config.Name or "Tab",
        TextColor3 = theme.TextDim,
        Font     = Enum.Font.GothamSemibold,
        TextSize = 13,
        Size     = UDim2.new(1, -(config.Icon and 26 or 0), 1, 0),
        ZIndex   = 10,
    }, btnRow)

    -- Content page (scrollable)
    local page = Builder.ScrollFrame({
        Name             = "Page_" .. tabId,
        BackgroundTransparency = 1,
        Size             = UDim2.fromScale(1, 1),
        ScrollBarThickness = 4,
        ScrollBarImageColor3 = theme.ScrollBar,
        CanvasSize       = UDim2.fromOffset(0, 0),
        AutomaticCanvasSize = Enum.AutomaticSize.Y,
        ZIndex           = 6,
        Visible          = false,
    }, self._contentArea)
    Builder.ListLayout({ Padding = UDim.new(0, 0), SortOrder = Enum.SortOrder.LayoutOrder }, page)

    local function activateTab()
        -- Deactivate all
        for id, data in pairs(self._tabs) do
            local isActive = (id == tabId)
            data.page.Visible = isActive
            Utility.Tween(data.sideBtn, Utility.TI(0.18), {
                BackgroundColor3 = isActive and theme.SidebarActive or theme.Sidebar,
                BackgroundTransparency = isActive and 0 or 1,
            })
            Utility.Tween(data.labelEl, Utility.TI(0.18), {
                TextColor3 = isActive and theme.Text or theme.TextDim,
            })
            if data.iconEl then
                local prop = data.iconEl:IsA("TextLabel") and "TextColor3" or "ImageColor3"
                Utility.Tween(data.iconEl, Utility.TI(0.18), {
                    [prop] = isActive and theme.AccentText or theme.TextDim,
                })
            end
        end
        self._activeTab = tabId

        -- Stagger-animate content on tab switch
        task.defer(function()
            local children = {}
            for _, child in ipairs(page:GetChildren()) do
                if child:IsA("Frame") then table.insert(children, child) end
            end
            Utility.Stagger(children, 0.03, function(f, i)
                f.BackgroundTransparency = f.BackgroundTransparency == 1 and 1 or 0.8
                Utility.Tween(f, Utility.TI(0.22, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                    BackgroundTransparency = 0,
                })
            end)
        end)
    end

    self._tabs[tabId] = {
        page     = page,
        sideBtn  = sideBtn,
        labelEl  = tabLabel,
        iconEl   = iconEl,
        config   = config,
    }

    -- Sidebar click
    local sideBtnClick = Builder.TextButton({
        Text="", BackgroundTransparency=1, Size=UDim2.fromScale(1,1), ZIndex=11,
    }, sideBtn)
    sideBtnClick.MouseButton1Click:Connect(function()
        activateTab()
    end)
    sideBtnClick.MouseEnter:Connect(function()
        if self._activeTab ~= tabId then
            Utility.Tween(sideBtn, Utility.TI(0.12), { BackgroundColor3 = theme.SidebarHover, BackgroundTransparency = 0 })
        end
    end)
    sideBtnClick.MouseLeave:Connect(function()
        if self._activeTab ~= tabId then
            Utility.Tween(sideBtn, Utility.TI(0.12), { BackgroundTransparency = 1 })
        end
    end)

    -- Auto activate first tab
    if not self._activeTab then
        activateTab()
    end

    -- ── Tab component API ──────────────────────────────────────────────────────
    local tab = {}
    local layoutOrder = 0

    local function nextOrder()
        layoutOrder = layoutOrder + 1
        return layoutOrder
    end

    local function makeComp(compName, hasApi)
        return function(self_tab, cfg)
            cfg = cfg or {}
            cfg._layoutOrder = nextOrder()
            local frame, api = Components[compName](page, cfg, theme, tabId)
            frame.LayoutOrder = cfg._layoutOrder
            if cfg.Tooltip then
                TooltipSystem.Attach(frame, cfg.Tooltip, theme)
            end
            return api or {}
        end
    end

    tab.CreateButton      = makeComp("Button",      true)
    tab.CreateToggle      = makeComp("Toggle",      true)
    tab.CreateSlider      = makeComp("Slider",      true)
    tab.CreateDropdown    = makeComp("Dropdown",    true)
    tab.CreateTextbox     = makeComp("Textbox",     true)
    tab.CreateKeybind     = makeComp("Keybind",     true)
    tab.CreateLabel       = makeComp("Label",       true)
    tab.CreateParagraph   = makeComp("Paragraph",   true)
    tab.CreateColorPicker = makeComp("ColorPicker", true)
    tab.CreateProgressBar = makeComp("ProgressBar", true)
    tab.CreateAccordion   = makeComp("Accordion",   true)
    tab.CreateRadioGroup  = makeComp("RadioGroup",  true)
    tab.CreateDivider     = makeComp("Divider",     false)

    function tab:CreateSection(cfg)
        cfg = cfg or {}
        cfg._layoutOrder = nextOrder()
        local frame = Components.Section(page, cfg, theme)
        frame.LayoutOrder = cfg._layoutOrder
        return {}
    end

    return tab
end

-- ── Window APIs ───────────────────────────────────────────────────────────────
function Window:Show()
    self._visible = true
    self._minimized = false
    self._root.Visible = true
    self._restoreBtn.Visible = false
    Utility.Tween(self._root, Utility.TI(0.3, Enum.EasingStyle.Back), { Size = self._root.Size })
end

function Window:Hide()
    self._visible = false
    Utility.Tween(self._root, Utility.TI(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
        BackgroundTransparency = 1,
    })
    task.delay(0.22, function()
        if self._root and self._root.Parent then
            self._root.Visible = false
            self._root.BackgroundTransparency = self._theme.Transparency or 0
        end
    end)
end

function Window:Minimize()
    self._minimized = true
    local fullSize = self._root.Size
    self._fullSize = fullSize
    Utility.Tween(self._root, Utility.TI(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
        Size = UDim2.fromOffset(fullSize.X.Offset, 0),
    })
    task.delay(0.32, function()
        if self._root and self._root.Parent then
            self._root.Visible = false
            self._restoreBtn.Visible = true
        end
    end)
end

function Window:Restore()
    self._minimized = false
    self._visible   = true
    self._restoreBtn.Visible = false
    self._root.Visible = true
    local targetSize = self._fullSize or self._root.Size
    self._root.Size = UDim2.fromOffset(targetSize.X.Offset, 0)
    Utility.Tween(self._root, Utility.TI(0.35, Enum.EasingStyle.Back), { Size = targetSize })
end

function Window:Toggle()
    if self._visible then
        self:Hide()
    else
        self:Show()
    end
end

function Window:Notify(config)
    NotificationSystem.Send(config)
end

function Window:Destroy()
    self._connections:Clean()
    if self._toggleConns then self._toggleConns:Clean() end
    if self._toggleBtn and self._toggleBtn.Parent then
        pcall(function() self._toggleBtn:Destroy() end)
    end
    if self._restoreBtn and self._restoreBtn.Parent then
        pcall(function() self._restoreBtn:Destroy() end)
    end
    if self._root and self._root.Parent then
        pcall(function() self._root:Destroy() end)
    end
end

function Mixusfield:_Init()
    if self._initialized then return end
    self._initialized = true

    local existing = CoreGui:FindFirstChild("MixusfieldGui")
    if existing then
        pcall(function() existing:Destroy() end)
    end

    local sg = Instance.new("ScreenGui")
    sg.Name            = "MixusfieldGui"
    sg.ResetOnSpawn    = false
    sg.IgnoreGuiInset  = true
    sg.ZIndexBehavior  = Enum.ZIndexBehavior.Global
    sg.DisplayOrder    = 999

    local ok = pcall(function()
        sg.Parent = CoreGui
    end)
    if not ok then
        local playerGui = LocalPlayer:WaitForChild("PlayerGui", 10)
        if playerGui then
            sg.Parent = playerGui
        else
            warn("[Mixusfield] Could not find PlayerGui – ScreenGui may not display.")
            sg.Parent = game:GetService("CoreGui")
        end
    end
    self._screenGui = sg

    local theme = ThemeManager.Get()
    NotificationSystem.Init(self._screenGui, theme)
    TooltipSystem.Init(self._screenGui)
end

function Mixusfield:CreateWindow(config)
    self:_Init()
    if not self._screenGui then
        warn("[Mixusfield] CreateWindow: ScreenGui failed to initialize.")
        return setmetatable({}, { __index = function() return function() return {} end end })
    end

    config = config or {}
    local theme = ThemeManager.Set(config.Theme or "Dark", config.Accent)

    -- ── No key system ──────────────────────────────────────────────────────────
    if not config.KeySystem then
        local win = Window.new(self._screenGui, config, theme)
        table.insert(self._windows, win)
        return win
    end

    -- ── Key system: proxy pattern ──────────────────────────────────────────────
    local windowProxy = {}
    local realWindow  = nil
    local tabQueue    = {}
    local resolved    = false

    local function makeTabProxy(queuedEntry)
        local tabProxy = {}
        local tabReal  = nil
        local compQueue = {}

        local function replayComp()
            for _, call in ipairs(compQueue) do
                local m    = call[1]
                local args = call[2]
                if tabReal and tabReal[m] then
                    tabReal[m](tabReal, table.unpack(args))
                end
            end
            compQueue = {}
        end

        queuedEntry._tabProxy = tabProxy
        queuedEntry._replayFn = function(realTab)
            tabReal = realTab
            replayComp()
        end

        local compMethods = {
            "CreateButton","CreateToggle","CreateSlider","CreateDropdown",
            "CreateTextbox","CreateKeybind","CreateLabel","CreateParagraph",
            "CreateSection","CreateColorPicker","CreateProgressBar",
            "CreateAccordion","CreateRadioGroup","CreateDivider"
        }
        for _, mName in ipairs(compMethods) do
            tabProxy[mName] = function(self_proxy, cfg)
                if tabReal then
                    return tabReal[mName](tabReal, cfg)
                else
                    table.insert(compQueue, {mName, {cfg}})
                    local apiProxy = {}
                    setmetatable(apiProxy, { __index = function() return function() end end })
                    return apiProxy
                end
            end
        end

        return tabProxy
    end

    local windowMeta = {}
    windowMeta.__index = function(_, k)
        if k == "CreateTab" then
            return function(self_proxy, cfg)
                if realWindow then
                    return realWindow:CreateTab(cfg)
                end
                local entry = { method = "CreateTab", args = {cfg} }
                table.insert(tabQueue, entry)
                return makeTabProxy(entry)
            end
        end
        return function(self_proxy, ...)
            if realWindow then
                local fn = realWindow[k]
                if type(fn) == "function" then return fn(realWindow, ...) end
            else
                local args = {...}
                table.insert(tabQueue, { method = k, args = args, isGeneric = true })
            end
        end
    end
    setmetatable(windowProxy, windowMeta)

    local function onSuccess()
        realWindow = Window.new(self._screenGui, config, theme)
        table.insert(self._windows, realWindow)
        resolved = true

        for _, entry in ipairs(tabQueue) do
            if entry.isGeneric then
                local fn = realWindow[entry.method]
                if type(fn) == "function" then
                    pcall(fn, realWindow, table.unpack(entry.args))
                end
            elseif entry.method == "CreateTab" then
                local realTab = realWindow:CreateTab(entry.args[1])
                if entry._replayFn then
                    entry._replayFn(realTab)
                end
            end
        end
        tabQueue = {}
    end

    KeySystem.Create(self._screenGui, config.KeySystem, theme, onSuccess)
    return windowProxy
end

function Mixusfield:Notify(config)
    self:_Init()
    if not self._screenGui then return end
    NotificationSystem.Send(config)
end

-- SetTheme: full repaint for sidebar window layout
function Mixusfield:SetTheme(themeName, accent)
    local theme = ThemeManager.Set(themeName, accent)

    for _, win in ipairs(self._windows) do
        pcall(function()
            win._theme = theme

            local root    = win._root
            local topBar  = win._topBar
            local sidebar = win._sidebar
            local content = win._contentArea

            if not (root and root.Parent) then return end

            -- Root
            Utility.Tween(root, Utility.TI(0.3), {
                BackgroundColor3       = theme.Background,
                BackgroundTransparency = theme.Transparency,
            })
            for _, v in ipairs(root:GetChildren()) do
                if v:IsA("UIStroke") then v.Color = theme.Border end
            end

            -- Top bar
            if topBar then
                Utility.Tween(topBar, Utility.TI(0.3), { BackgroundColor3 = theme.TopBar })
                for _, v in ipairs(topBar:GetDescendants()) do
                    if v:IsA("Frame") and v.Size == UDim2.new(1,0,0,2) then
                        -- accent line
                        Utility.Tween(v, Utility.TI(0.3), { BackgroundColor3 = theme.Accent })
                    elseif v:IsA("Frame") and v.Size == UDim2.new(1,0,0,12) then
                        Utility.Tween(v, Utility.TI(0.3), { BackgroundColor3 = theme.TopBar })
                    end
                    if v:IsA("TextLabel") and v.Font == Enum.Font.GothamBold and v.TextSize == 13 and (v.Name ~= "Icon") then
                        Utility.Tween(v, Utility.TI(0.3), { TextColor3 = theme.AccentText })
                    end
                end
            end

            -- Sidebar
            if sidebar then
                Utility.Tween(sidebar, Utility.TI(0.3), { BackgroundColor3 = theme.Sidebar })
                -- Sidebar right border
                for _, v in ipairs(sidebar:GetChildren()) do
                    if v:IsA("Frame") and v.Size == UDim2.new(0,1,1,0) then
                        v.BackgroundColor3 = theme.Border
                    end
                end
                -- Search container
                for _, v in ipairs(sidebar:GetDescendants()) do
                    if v:IsA("Frame") and v.Name ~= "MixusfieldWindow_" .. "" then
                        -- sidebar tab buttons
                    end
                    if v:IsA("UIStroke") then v.Color = theme.Border end
                    if v:IsA("TextBox") then
                        v.TextColor3 = theme.Text
                        v.PlaceholderColor3 = theme.TextMuted
                    end
                end
                -- Update tab buttons in sidebar
                for id, data in pairs(win._tabs) do
                    local isActive = (id == win._activeTab)
                    Utility.Tween(data.sideBtn, Utility.TI(0.3), {
                        BackgroundColor3       = isActive and theme.SidebarActive or theme.Sidebar,
                        BackgroundTransparency = isActive and 0 or 1,
                    })
                    Utility.Tween(data.labelEl, Utility.TI(0.3), {
                        TextColor3 = isActive and theme.Text or theme.TextDim,
                    })
                    if data.iconEl then
                        local prop = data.iconEl:IsA("TextLabel") and "TextColor3" or "ImageColor3"
                        Utility.Tween(data.iconEl, Utility.TI(0.3), {
                            [prop] = isActive and theme.AccentText or theme.TextDim,
                        })
                    end
                end
            end

            -- Content pages
            if content then
                for _, page in ipairs(content:GetChildren()) do
                    if page:IsA("ScrollingFrame") then
                        page.ScrollBarImageColor3 = theme.ScrollBar

                        for _, comp in ipairs(page:GetDescendants()) do
                            local n = comp.Name or ""

                            if comp:IsA("Frame") then
                                if n:sub(1,7)  == "Button_"   or n:sub(1,7)  == "Toggle_"
                                or n:sub(1,7)  == "Slider_"   or n:sub(1,9)  == "Dropdown_"
                                or n:sub(1,8)  == "Textbox_"  or n:sub(1,8)  == "Keybind_"
                                or n:sub(1,10) == "Paragraph_" or n:sub(1,12) == "ProgressBar_"
                                or n:sub(1,12) == "ColorPicker_" or n:sub(1,10) == "Accordion_"
                                or n:sub(1,11) == "RadioGroup_" or n:sub(1,6)  == "Label_" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.Surface })
                                elseif n:sub(1,8) == "Section_" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.Surface })
                                elseif n:sub(1,8) == "Divider_" then
                                    -- transparent, no repaint
                                elseif n == "TogglePill" then
                                    if comp.BackgroundColor3 ~= theme.Accent then
                                        Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.ToggleOff })
                                    end
                                elseif n == "SliderTrack" or n == "ProgressTrack" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.SliderTrack })
                                elseif n == "SliderFill" or n == "ProgressFill" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.Accent })
                                elseif n == "KeyPill" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.SurfaceAlt })
                                elseif n == "InputContainer" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.InputBackground })
                                elseif n == "AccentBar" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.Accent })
                                elseif n == "DropPanel" or n == "ColorPickerPanel" then
                                    Utility.Tween(comp, Utility.TI(0.3), { BackgroundColor3 = theme.SurfaceAlt })
                                end

                                for _, s in ipairs(comp:GetChildren()) do
                                    if s:IsA("UIStroke") then
                                        if n == "SliderThumb" then
                                            s.Color = theme.Accent
                                        else
                                            s.Color = theme.Border
                                        end
                                    end
                                end
                            end

                            if comp:IsA("TextLabel") then
                                local parName = (comp.Parent and comp.Parent.Name) or ""
                                if parName == "KeyPill" then
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.AccentText })
                                elseif comp.TextXAlignment == Enum.TextXAlignment.Right and comp.TextSize == 12 then
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.AccentText })
                                elseif comp.TextSize <= 11 then
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.TextMuted })
                                elseif n == "Icon" then
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.TextDim })
                                elseif comp.Font == Enum.Font.GothamBold and comp.TextSize >= 13 then
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.Text })
                                else
                                    Utility.Tween(comp, Utility.TI(0.3), { TextColor3 = theme.TextDim })
                                end
                            end

                            if comp:IsA("ImageLabel") and n == "Icon" then
                                Utility.Tween(comp, Utility.TI(0.3), { ImageColor3 = theme.TextDim })
                            end

                            if comp:IsA("TextBox") then
                                comp.TextColor3        = theme.Text
                                comp.PlaceholderColor3 = theme.TextMuted
                            end

                            if comp:IsA("ScrollingFrame") then
                                comp.ScrollBarImageColor3 = theme.ScrollBar
                            end
                        end
                    end
                end
            end

            -- Search bar
            if win._searchField then
                win._searchField.TextColor3        = theme.Text
                win._searchField.PlaceholderColor3 = theme.TextMuted
            end

            -- Restore button
            if win._restoreBtn then
                Utility.Tween(win._restoreBtn, Utility.TI(0.3), { BackgroundColor3 = theme.Surface })
                for _, v in ipairs(win._restoreBtn:GetDescendants()) do
                    if v:IsA("UIStroke") then v.Color = theme.Border end
                    if v:IsA("TextLabel") and v.Name ~= "Icon" then
                        Utility.Tween(v, Utility.TI(0.3), { TextColor3 = theme.Text })
                    end
                    if v:IsA("TextLabel") and v.Name == "Icon" then
                        Utility.Tween(v, Utility.TI(0.3), { TextColor3 = theme.AccentText })
                    end
                end
            end
        end) -- end pcall
    end

    return theme
end

function Mixusfield:Alert(config)
    self:_Init()
    if not self._screenGui then return end
    local theme = ThemeManager.Get()
    return AlertSystem.Show(self._screenGui, config, theme)
end

function Mixusfield:Destroy()
    for _, win in ipairs(self._windows) do
        pcall(function() win:Destroy() end)
    end
    self._windows = {}
    SearchRegistry.Clear()
    if self._screenGui then
        pcall(function() self._screenGui:Destroy() end)
        self._screenGui = nil
    end
    self._initialized = false
end
