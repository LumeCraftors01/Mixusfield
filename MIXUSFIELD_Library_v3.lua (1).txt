-- ============================================================
-- /// SAVE SYSTEM
-- Wraps writefile/readfile with JSON encoding.
-- Falls back silently if the executor doesn't support files.
-- ============================================================
local SaveSystem = {}
SaveSystem._data  = {}
SaveSystem._file  = "MixusfieldSave.json"

function SaveSystem:_load()
  local ok, raw = pcall(readfile, self._file)
  if ok and raw and #raw > 0 then
    local jok, parsed = pcall(HttpService.JSONDecode, HttpService, raw)
    if jok and type(parsed) == "table" then
      self._data = parsed
    end
  end
end

function SaveSystem:_flush()
  local ok, encoded = pcall(HttpService.JSONEncode, HttpService, self._data)
  if ok then
    pcall(writefile, self._file, encoded)
  end
end

function SaveSystem:Get(key, default)
  return self._data[key] ~= nil and self._data[key] or default
end

function SaveSystem:Set(key, value)
  self._data[key] = value
  self:_flush()
end

function SaveSystem:Clear(key)
  if key then
    self._data[key] = nil
  else
    self._data = {}
  end
  self:_flush()
end

function SaveSystem:SetFile(filename)
  self._file = filename
  self:_load()
end

-- Boot-load the save file right away
SaveSystem:_load()

local ThemeListeners = {}

local function ThemeBind(inst, prop, key)
  table.insert(ThemeListeners, { Instance = inst, Property = prop, Key = key })
  pcall(function()
    if inst and inst.Parent then
      inst[prop] = Theme[key]
    end
  end)
end

local Custom = {}

function Custom:Create(Name, Properties, Parent)
  local inst = Instance.new(Name)
  for i, v in pairs(Properties) do
    pcall(function() inst[i] = v end)
  end
  if Parent then inst.Parent = Parent end
  return inst
end

local function MakeDraggable(topbar, object)
  local dragging, dragStart, startPos = false, nil, nil
  topbar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
      dragging  = true
      dragStart = input.Position
      startPos  = object.Position
      input.Changed:Connect(function()
        if input.UserInputState == Enum.UserInputState.End then
          dragging = false
        end
      end)
    end
  end)
  topbar.InputChanged:Connect(function(input)
    if dragging and (
      input.UserInputType == Enum.UserInputType.MouseMovement
      or input.UserInputType == Enum.UserInputType.Touch
    ) then
      local delta = input.Position - dragStart
      object.Position = UDim2.new(
        startPos.X.Scale, startPos.X.Offset + delta.X,
        startPos.Y.Scale, startPos.Y.Offset + delta.Y
      )
    end
  end)
end

-- /// THEME SYSTEM – PUBLIC API
-- ============================================================
function MIXUSFIELD_Library:SetTheme(nameOrTable)
  local newTheme
  if type(nameOrTable) == "string" then
    for k, v in pairs(THEMES) do
      if k:lower() == nameOrTable:lower() then
        newTheme = v; break
      end
    end
    if not newTheme then
      warn("[MIXUSFIELD] Unknown theme: " .. tostring(nameOrTable))
      return
    end
  elseif type(nameOrTable) == "table" then
    newTheme = nameOrTable
  else
    warn("[MIXUSFIELD] SetTheme expects a string or color table.")
    return
  end

  for k, v in pairs(newTheme) do Theme[k] = v end

  local tweenInfo = TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
  for _, entry in ipairs(ThemeListeners) do
    pcall(function()
      if entry.Instance and entry.Instance.Parent then
        if typeof(entry.Instance[entry.Property]) == "Color3" then
          TweenService:Create(entry.Instance, tweenInfo, {
            [entry.Property] = Theme[entry.Key]
          }):Play()
        else
          entry.Instance[entry.Property] = Theme[entry.Key]
        end
      end
    end)
  end
end

function MIXUSFIELD_Library:GetThemeNames()
  local names = {}
  for k in pairs(THEMES) do table.insert(names, k) end
  table.sort(names)
  return names
end

function MIXUSFIELD_Library:RegisterTheme(name, colors)
  THEMES[name] = colors
end

  ThemeBind(ProgFill, "BackgroundColor3", "Accent")
  Custom:Create("UICorner", { CornerRadius = UDim.new(0, 2) }, ProgFill)

  task.wait()
  local bodyLines = math.max(1, math.ceil(BodyLabel.TextBounds.X / math.max(BodyLabel.AbsoluteSize.X, 1)))
  BodyLabel.Size  = UDim2.new(1, -16, 0, 14 * bodyLines)
  local cardH     = BodyLabel.AbsolutePosition.Y - Card.AbsolutePosition.Y + BodyLabel.AbsoluteSize.Y + 10

  -- Slide-in: start off right
  Card.Size        = UDim2.new(1, 0, 0, cardH)
  Card.Position    = UDim2.new(1.3, 0, 0, 0)  -- dummy; layout handles Y
  Card.AnchorPoint = Vector2.new(0, 0)

  -- We do an X offset instead via ClipsDescendants on layout wrapper

      if d:IsA("TextLabel") or d:IsA("TextButton") then
        d.TextTransparency = 1
      elseif d:IsA("ImageLabel") then
        d.ImageTransparency = 1
      elseif d:IsA("Frame") then
        d.BackgroundTransparency = 1
      elseif d:IsA("UIStroke") then
        d.Transparency = 1
      end
    end)
  end

  local fadeIn = TweenInfo.new(Time, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
  TweenService:Create(Card, fadeIn, { BackgroundTransparency = 0 }):Play()
  for _, d in ipairs(Card:GetDescendants()) do
    pcall(function()
      if d:IsA("TextLabel") then
        TweenService:Create(d, fadeIn, { TextTransparency = 0 }):Play()
      elseif d:IsA("TextButton") then
        TweenService:Create(d, fadeIn, { TextTransparency = 0 }):Play()
      elseif d:IsA("ImageLabel") then
        TweenService:Create(d, fadeIn, { ImageTransparency = 0 }):Play()
      elseif d:IsA("Frame") and d.Name ~= "AccentBar" then
        local target = d.Name == "ProgBg" and 0.3 or 0
        TweenService:Create(d, fadeIn, { BackgroundTransparency = target }):Play()
      elseif d:IsA("UIStroke") then
        TweenService:Create(d, fadeIn, { Transparency = 0 }):Play()
      end
    end)
  end

  local Notif    = {}
  local dismissed = false

  function Notif:Close()
    if dismissed then return end
    dismissed = true
    local fadeOut = TweenInfo.new(Time * 0.7, Enum.EasingStyle.Quad, Enum.EasingDirection.In)
    TweenService:Create(Card, fadeOut, { BackgroundTransparency = 1 }):Play()
    for _, d in ipairs(Card:GetDescendants()) do
      pcall(function()
        if d:IsA("TextLabel") then
          TweenService:Create(d, fadeOut, { TextTransparency = 1 }):Play()
        elseif d:IsA("TextButton") then
          TweenService:Create(d, fadeOut, { TextTransparency = 1 }):Play()
        elseif d:IsA("ImageLabel") then
          TweenService:Create(d, fadeOut, { ImageTransparency = 1 }):Play()
        elseif d:IsA("Frame") then
          TweenService:Create(d, fadeOut, { BackgroundTransparency = 1 }):Play()
        elseif d:IsA("UIStroke") then
          TweenService:Create(d, fadeOut, { Transparency = 1 }):Play()
        end
      end)
    end
    task.wait(Time)
    TweenService:Create(Card, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut), {
      Size = UDim2.new(1, 0, 0, 0)
    }):Play()
    task.wait(0.25)
    Card:Destroy()
  end

  CloseBtn.Activated:Connect(function() Notif:Close() end)

  -- Animate the timer bar and auto-close
  task.spawn(function()
    TweenService:Create(ProgFill, TweenInfo.new(Delay, Enum.EasingStyle.Linear), {
      Size = UDim2.new(0, 0, 1, 0)
    }):Play()
    task.wait(Delay)
    Notif:Close()
  end)

  return Notif
end

-- ============================================================
-- /// KEY SYSTEM  v2
-- Config: Key, Title, SubTitle, Note, Callback, SaveKey, Expiry
-- SaveKey (string) – if set, accepted key is remembered in save file
-- Expiry  (number) – seconds until saved key expires (0 = forever)
-- ============================================================
function MIXUSFIELD_Library:KeySystem(Config, onSuccess)
  local ValidKey  = Config.Key      or ""
  local Title     = Config.Title    or "MIXUSFIELD"
  local SubTitle  = Config.SubTitle or "Key Required"
  local Note      = Config.Note     or "Enter the key below to continue."
  local Callback  = Config.Callback or function() end
  local SKey      = Config.SaveKey  -- save-data key for remembered entry
  local Expiry    = tonumber(Config.Expiry) or 0  -- 0 = never expires

  -- Check saved key first
  if SKey then
    local saved = SaveSystem:Get(SKey .. "_key")
    local savedAt = SaveSystem:Get(SKey .. "_savedAt", 0)
    local valid = saved == ValidKey
    if valid and Expiry > 0 then
      valid = (os.time() - savedAt) < Expiry
    end
    if valid then
      Callback()
      if type(onSuccess) == "function" then onSuccess() end
      return { Close = function() end }
    end
  end

  local KeyGui = Custom:Create("ScreenGui", {
    ZIndexBehavior = Enum.ZIndexBehavior.Sibling,
    Name           = "MixusKeyGui",

  local function TryKey()
    local entered = KeyInput.Text
    if entered == ValidKey then
      StatusLabel.TextColor3 = Color3.fromRGB(60, 210, 100)
      StatusLabel.Text       = "✓ Accepted! Loading..."
      TweenService:Create(InputStroke, TweenInfo.new(0.2), { Color = Color3.fromRGB(60, 210, 100) }):Play()
      if SKey then
        SaveSystem:Set(SKey .. "_key",    entered)
        SaveSystem:Set(SKey .. "_savedAt", os.time())
      end
      task.wait(0.8)
      Callback()
      KeyFuncs:Close()
      if type(onSuccess) == "function" then onSuccess() end
    else
      StatusLabel.TextColor3 = Color3.fromRGB(220, 60, 60)
      StatusLabel.Text       = "✗ Incorrect key."
      TweenService:Create(InputStroke, TweenInfo.new(0.1), { Color = Color3.fromRGB(220, 60, 60) }):Play()
      task.wait(0.5)
      TweenService:Create(InputStroke, TweenInfo.new(0.2), { Color = Theme.Stroke }):Play()
      StatusLabel.Text = ""
    end
  end

    CircleClick(SubmitBtn, Player:GetMouse().X, Player:GetMouse().Y)
    TryKey()
  end)
  KeyInput.FocusLost:Connect(function(enter)
    if enter then TryKey() end
  end)

  return KeyFuncs
end

        local height = 0
        for _, v in pairs(SectionAdd:GetChildren()) do
          if v.Name ~= "UIListLayout" then
            height = height + v.Size.Y.Offset + 3
          end

        TweenService:Create(SectionAdd, TweenInfo.new(0.12), { Size = UDim2.new(1, 0, 0, math.max(height, 0)) }):Play()
        TweenService:Create(Section,    TweenInfo.new(0.12), { Size = UDim2.new(1, 0, 0, 42 + math.max(height, 0)) }):Play()
        task.wait(0.2)
        UpdateSizeScroll()

        ThemeBind(SliderFill, "BackgroundColor3", "Accent")

        local Handle = Custom:Create("Frame", {

        local function RoundTo(n, factor)
          return math.floor(n / factor + 0.5) * factor
        end

        function SFuncs:Set(value)
          value        = math.clamp(RoundTo(value, Increment), SMin, SMax)
          SFuncs.Value = value
          TextBox.Text = tostring(value)
          TweenService:Create(SliderFill, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {

          if inp.UserInputType == Enum.UserInputType.MouseButton1

            Dragging = true
            -- Snap to click position immediately
            local scale = math.clamp((inp.Position.X - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1)
            SFuncs:Set(SMin + (SMax - SMin) * scale)
          end

          if inp.UserInputType == Enum.UserInputType.MouseButton1
          or inp.UserInputType == Enum.UserInputType.Touch then
            Dragging = false
            SCallback(SFuncs.Value)
          end
        end)

        local lastX = nil
        UserInputService.InputChanged:Connect(function(inp)
          if Dragging then
            local x = inp.Position.X
            if x ~= lastX then
              lastX = x
              local scale = math.clamp((x - SliderTrack.AbsolutePosition.X) / SliderTrack.AbsoluteSize.X, 0, 1)
              SFuncs:Set(SMin + (SMax - SMin) * scale)
            end
          end
        end)

          local clean = TextBox.Text:gsub("[^%d%.%-]", "")
          if clean ~= TextBox.Text then TextBox.Text = clean end
        end)
        TextBox.FocusLost:Connect(function()
          local v = tonumber(TextBox.Text) or SFuncs.Value
          SFuncs:Set(v)
          SCallback(SFuncs.Value)
        end)

        row.Size = UDim2.new(1, 0, 0, 52)
        SFuncs:Set(SDefault)
        SCallback(SFuncs.Value)
        task.defer(UpdateSizeSection)

        ItemCount += 1
        return SFuncs
      end

      function Item:AddKeybind(Config)
        local KTitle    = Config[1] or Config.Title    or ""
        local KContent  = Config[2] or Config.Content  or ""
        local KDefault  = Config[3] or Config.Default  or Enum.KeyCode.Unknown
        local KCallback = Config[4] or Config.Callback or function() end
        local KSaveKey  = Config[5] or Config.SaveKey  or nil
        local KFuncs    = { Value = KDefault }

        if KSaveKey then
          local saved = SaveSystem:Get(KSaveKey)
          if saved then
            local ok, kc = pcall(function() return Enum.KeyCode[saved] end)
            if ok and kc then KDefault = kc end
          end
          KFuncs.Value = KDefault
        end

        local row, stroke = MakeRow(36, "Keybind")
        MakeRowLabels(row, KTitle, KContent, 80)

        local KeyDisplay = Custom:Create("TextButton", {
          Font                   = Enum.Font.GothamBold,
          Text                   = tostring(KDefault.Name),
          TextColor3             = Theme.Accent,
          TextSize               = 11,
          AnchorPoint            = Vector2.new(1, 0.5),
          BackgroundColor3       = Theme.Accent,
          BackgroundTransparency = 0.88,
          BorderSizePixel        = 0,
          Position               = UDim2.new(1, -8, 0.5, 0),
          Size                   = UDim2.new(0, 70, 0, 22),
          Name                   = "KeyDisplay",
        }, row)
        ThemeBind(KeyDisplay, "BackgroundColor3", "Accent")
        ThemeBind(KeyDisplay, "TextColor3", "Accent")
        Custom:Create("UICorner", { CornerRadius = UDim.new(0, 5) }, KeyDisplay)

        local listening = false
        KeyDisplay.Activated:Connect(function()
          if listening then return end
          listening = true
          KeyDisplay.Text = "[ ... ]"
          TweenService:Create(stroke, TweenInfo.new(0.1), { Color = Theme.Accent, Transparency = 0.15 }):Play()

          local conn
          conn = UserInputService.InputBegan:Connect(function(inp, gp)
            if gp then return end
            if inp.UserInputType == Enum.UserInputType.Keyboard then
              KFuncs:Set(inp.KeyCode)
              conn:Disconnect()
              listening = false
              TweenService:Create(stroke, TweenInfo.new(0.1), { Color = Theme.Stroke, Transparency = 0.5 }):Play()
            end
          end)
        end)

        -- Fire callback when key is pressed in game
        UserInputService.InputBegan:Connect(function(inp, gp)
          if not gp and not listening and inp.KeyCode == KFuncs.Value then
            KCallback(KFuncs.Value)
          end
        end)

        function KFuncs:Set(keyCode)
          KFuncs.Value    = keyCode
          KeyDisplay.Text = keyCode.Name
          if KSaveKey then SaveSystem:Set(KSaveKey, keyCode.Name) end
          KCallback(keyCode)
        end

        KFuncs:Set(KDefault)
        ItemCount += 1
        return KFuncs
      end

          TweenService:Create(SelectStroke, TweenInfo.new(0.12), { Color = Theme.Stroke, Transparency = 0.4 }):Play()
        end)

        DFuncs:Refresh(DFuncs.Options, DFuncs.Value)

        ItemCount += 1
        CountDropdown += 1
        return DFuncs
      end -- AddDropdown

      -- ── ThemePicker (convenience helper) ────────────────────
      -- Adds a dropdown pre-filled with all available themes.
      function Item:AddThemePicker(Config)
        Config = Config or {}
        Config[1]  = Config[1] or Config.Title   or "Theme"
        Config[2]  = Config[2] or Config.Content or "Choose a color theme"
        Config[3]  = false  -- not multi
        Config[4]  = MIXUSFIELD_Library:GetThemeNames()
        Config[5]  = {}
        Config[7]  = Config.SaveKey
        local saved = Config[7] and SaveSystem:Get(Config[7])
        local initialTheme = saved or "Crimson"

        local df = Item:AddDropdown({
          Title    = Config[1],
          Content  = Config[2],
          Multi    = false,
          Options  = MIXUSFIELD_Library:GetThemeNames(),
          Default  = { initialTheme },
          SaveKey  = Config[7],
          Callback = function(val)
            if val and val[1] then
              MIXUSFIELD_Library:SetTheme(val[1])
            end
          end,
        })
        MIXUSFIELD_Library:SetTheme(initialTheme)
        return df
      end

      CountSection += 1
      return Item
    end -- AddSection

    CountTab += 1
    return Sections
  end -- CreateTab

  -- Expose window Funcs
  function Funcs:SetTitle(text)
    TitleLabel.Text = text
  end
  function Funcs:SetDesc(text)
    DescLabel.Text = text
  end
  function Funcs:Close()
    MIXUSGui:Destroy()
    MIXUSFIELD_Library.Unloaded = true
  end

  return Tabs
end -- CreateWindow
